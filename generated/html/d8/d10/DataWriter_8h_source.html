165template <class T, template<class, class...> class C, class... Args, typename A, typename B>
166void WriteSorted(const C<T, Args...> &container, A sortFn, B writeFn)
167{
168 std::vector<const T *> sorted;
169 sorted.reserve(container.size());
170 for(const auto &it : container)
171 sorted.emplace_back(&it);
172 std::sort(sorted.begin(), sorted.end(), sortFn);
173
174 for(const auto &sit : sorted)
175 writeFn(*sit);
176}

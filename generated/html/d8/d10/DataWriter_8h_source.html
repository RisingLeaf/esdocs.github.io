29// This class writes data in a hierarchical format, where an indented line is
30// considered the "child" of the first line above it that is less indented. By
31// using this class, you can have a function add data to the file without having
32// to tell that function what indentation level it is at. This class also
33// automatically adds quotation marks around strings if they contain whitespace.
34class DataWriter {
35public:
36 // Constructor, specifying the file to write.
37 explicit DataWriter(const std::string &path);
38 // Constructor for a DataWriter that will not save its contents automatically
39 DataWriter();
40 DataWriter(const DataWriter &) = delete;
41 DataWriter(DataWriter &&) = delete;
42 DataWriter &operator=(const DataWriter &) = delete;
43 DataWriter operator=(DataWriter &&) = delete;
44 // The file is not actually saved until the destructor is called. This makes
45 // it possible to write the whole file in a single chunk.
46 ~DataWriter();
47
48 // Save the contents to a file.
49 void SaveToPath(const std::string &path);
50
51 // The Write() function can take any number of arguments. Each argument is
52 // converted to a token. Arguments may be strings or numeric values.
53 template <class A, class ...B>
54 void Write(const A &a, B... others);
55 // Write the entire structure represented by a DataNode, including any
56 // children that it has.
57 void Write(const DataNode &node);
58 // End the current line. This can be used to add line breaks or to terminate
59 // a line you have been writing token by token with WriteToken().
60 void Write();
61
62 // Begin a new line that is a "child" of the previous line.
63 void BeginChild();
64 // Finish writing a block of child nodes and decrease the indentation.
65 void EndChild();
66
67 // Write a comment. It will be at the current indentation level, and will
68 // have "# " inserted before it.
69 void WriteComment(const std::string &str);
70
71 // Write a token, without writing a whole line. Use this very carefully.
72 void WriteToken(const char *a);
73 void WriteToken(const std::string &a);
74 // Write a token of any arithmetic type.
75 template <class A>
76 void WriteToken(const A &a);
77
78
79private:
80 // Save path (in UTF-8). Empty string for in-memory DataWriter.
81 std::string path;
82 // Current indentation level.
83 std::string indent;
84 // Before writing each token, we will write either the indentation string
85 // above, or this string.
86 static const std::string space;
87 // Remember which string should be written before the next token. This is
88 // "indent" for the first token in a line and "space" for subsequent tokens.
89 const std::string *before;
90 // Compose the output in memory before writing it to file.
91 std::ostringstream out;
92};

96// The Write() function can take any number of arguments, each of which becomes
97// a token. They must be either strings or numeric types.
98template <class A, class ...B>
99void DataWriter::Write(const A &a, B... others)
100{
101 WriteToken(a);
102 Write(others...);

107// Write any numeric type as a single token.
108template <class A>
109void DataWriter::WriteToken(const A &a)
110{
111 static_assert(std::is_arithmetic_v<A>,
112 "DataWriter cannot output anything but strings and arithmetic types.");
113
114 out << *before << a;
115 before = &space;
116}

120// Encapsulate the logic for writing the contents of a collection in a sorted manner. The caller
121// should provide a sorting method; it will be called with pointers to the type of the container.
122// The provided write method will be called for each element of the container.
123template <class T, template<class, class...> class C, class... Args, typename A, typename B>
124void WriteSorted(const C<T, Args...> &container, A sortFn, B writeFn)
125{
126 std::vector<const T *> sorted;
127 sorted.reserve(container.size());
128 for(const auto &it : container)
129 sorted.emplace_back(&it);
130 std::sort(sorted.begin(), sorted.end(), sortFn);
131
132 for(const auto &sit : sorted)
133 writeFn(*sit);
134}
135template <class K, class V, class... Args, typename A, typename B>
136void WriteSorted(const std::map<const K *, V, Args...> &container, A sortFn, B writeFn)
137{
138 std::vector<const std::pair<const K *const, V> *> sorted;
139 sorted.reserve(container.size());

141 sorted.emplace_back(&it);
142 std::sort(sorted.begin(), sorted.end(), sortFn);
143
144 for(const auto &sit : sorted)
145 writeFn(*sit);
146}

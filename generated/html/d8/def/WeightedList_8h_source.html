31// Template representing a list of objects of a given type where each item in the
32// list is weighted with an integer. This list can be queried to randomly return
33// one object from the list where the probability of an object being returned is
34// the weight of the object over the sum of the weights of all objects in the list.
35template <class Type>
36class WeightedList {
37 using iterator = typename std::vector<Type>::iterator;
38 using const_iterator = typename std::vector<Type>::const_iterator;
39public:
40 template <class T>
41 friend std::size_t erase(WeightedList<T> &list, const T &item);
42 template <class T, class UnaryPredicate>
43 friend std::size_t erase_if(WeightedList<T> &list, UnaryPredicate pred);
44
45 const Type &Get() const;
46 std::size_t TotalWeight() const noexcept { return total; }

48 // Average the result of the given function by the choices' weights.
49 template <class Callable>
50 std::enable_if_t<

52 // The return type of WeightedList::Average, if the above test passes:
53 std::invoke_result_t<Callable&&, const Type&&>
54 > Average(Callable c) const;
55 // Supplying a callable that does not return an arithmetic value will fail to compile.

57 iterator begin() noexcept { return choices.begin(); }
58 const_iterator begin() const noexcept { return choices.begin(); }
59 iterator end() noexcept { return choices.end(); }
60 const_iterator end() const noexcept { return choices.end(); }

96template <class T, class UnaryPredicate>
97std::size_t erase_if(WeightedList<T> &list, UnaryPredicate pred)
98{
99 std::size_t erased = 0;
100 unsigned available = list.choices.size() - 1;
101 for(unsigned index = 0; index < list.choices.size() - erased; ++index)
102 if(pred(list.choices[index]))
103 {
104 while(index != available && pred(list.choices[available]))
105 {
106 --available;
107 ++erased;
108 }
109 if(index != available)
110 {
111 list.choices[index] = std::move(list.choices[available]);
112 list.weights[index] = std::move(list.weights[available]);
113 --available;
114 }
115 ++erased;
116 }
117
118 list.choices.erase(std::prev(list.choices.end(), erased), list.choices.end());
119 list.weights.erase(std::prev(list.weights.end(), erased), list.weights.end());
120 list.RecalculateWeight();
121
122 return erased;
123}

127template <class Type>
128const Type &WeightedList<Type>::Get() const
129{
130 if(empty())
131 throw std::runtime_error("Attempted to call Get on an empty weighted list.");
132
133 unsigned index = 0;
134 for(unsigned choice = Random::Int(total); choice >= weights[index]; ++index)
135 choice -= weights[index];
136
137 return choices[index];
138}

142template <class Type>
143template <class Callable>
144std::enable_if_t<
145 std::is_arithmetic_v<std::invoke_result_t<Callable&&, const Type&&>>,
146 std::invoke_result_t<Callable&&, const Type&&>
147> WeightedList<Type>::Average(Callable fn) const
148{
149 std::size_t tw = TotalWeight();
150 if(tw == 0) return 0;
151
152 auto sum = typename std::invoke_result_t<Callable, const Type &>{};
153 for(unsigned index = 0; index < choices.size(); ++index)
154 sum += fn(choices[index]) * weights[index];
155 return sum / tw;
156}

160template <class Type>
161template <class ...Args>
162Type &WeightedList<Type>::emplace_back(int weight, Args&&... args)
163{
164 // All weights must be >= 1.
165 if(weight < 1)
166 throw std::invalid_argument("Invalid weight inserted into weighted list. Weights must be >= 1.");
167
168 choices.emplace_back(std::forward<Args>(args)...);
169 weights.emplace_back(weight);
170 total += weights.back();
171 return choices.back();
172}

176template <class Type>
177typename std::vector<Type>::iterator WeightedList<Type>::eraseAt(typename std::vector<Type>::iterator position) noexcept
178{
179 unsigned index = std::distance(choices.begin(), position);
180 total -= weights[index];
181 weights.erase(std::next(weights.begin(), index));
182 return choices.erase(position);
183}

187template <class Type>
188typename std::vector<Type>::iterator WeightedList<Type>::erase(typename std::vector<Type>::iterator first,
189 typename std::vector<Type>::iterator last) noexcept
190{
191 auto firstWeight = std::next(weights.begin(), std::distance(choices.begin(), first));
192 auto lastWeight = std::next(weights.begin(), std::distance(choices.begin(), last));
193 weights.erase(firstWeight, lastWeight);
194 RecalculateWeight();
195 return choices.erase(first, last);
196}

200template <class Type>
201void WeightedList<Type>::RecalculateWeight()
202{
203 total = std::accumulate(weights.begin(), weights.end(), 0);

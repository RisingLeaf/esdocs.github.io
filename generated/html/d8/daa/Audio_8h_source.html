27// This class is a collection of global functions for handling audio. A sound
28// can be played from any point in the code, and from any thread, just by
29// specifying the name of the sound to play. Most sounds will come from a
30// "source" at a certain position, and their volume and left / right balance is
31// adjusted based on how far they are from the observer. Sounds that are not
32// marked as looping will play once, then stop; looping sounds continue until
33// their source stops calling the "play" function for them.
34class Audio {
35public:
36 // Begin loading sounds (in a separate thread).
37 static void Init(const std::vector<std::string> &sources);
38 static void CheckReferences();
39
40 // Report the progress of loading sounds.
41 static double GetProgress();
42
43 // Get or set the volume (between 0 and 1).
44 static double Volume();
45 static void SetVolume(double level);
46
47 // Get a pointer to the named sound. The name is the path relative to the
48 // "sound/" folder, and without ~ if it's on the end, or the extension.
49 // Do not call this function until Progress() is 100%.
50 static const Sound *Get(const std::string &name);
51
52 // Set the listener's position, and also update any sounds that have been
53 // added but deferred because they were added from a thread other than the
54 // main one (the one that called Init()).
55 static void Update(const Point &listenerPosition);
56
57 // Play the given sound, at full volume.
58 static void Play(const Sound *sound);
59
60 // Play the given sound, as if it is at the given distance from the
61 // "listener". This will make it softer and change the left / right balance.
62 static void Play(const Sound *sound, const Point &position);
63
64 // Play the given music. An empty string means to play nothing.
65 static void PlayMusic(const std::string &name);
66
67 // Begin playing all the sounds that have been added since the last time
68 // this function was called.
69 static void Step();
70
71 // Shut down the audio system (because we're about to quit).
72 static void Quit();
73};

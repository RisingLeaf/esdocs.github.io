35// Class representing a UI window (full screen or pop-up) which responds to user
36// input and can draw itself. Everything displayed in the game is drawn in a
37// Panel, and panels can stack on top of each other like "real" UI windows. By
38// default, a panel allows the panels under it to show through, but does not
39// allow them to receive any events that it does not know how to handle.
40class Panel {
41public:
42 // Draw a sprite repeatedly to make a vertical edge.
43 static void DrawEdgeSprite(const Sprite *edgeSprite, int posX);
44

46public:

48 virtual ~Panel() = default;

50 // Move the state of this panel forward one game step.
51 virtual void Step();
52
53 // Draw this panel.
54 virtual void Draw() = 0;
55
56 // Return true if this is a full-screen panel, so there is no point in
57 // drawing any of the panels under it.
58 bool IsFullScreen() const noexcept;
59 // Return true if, when this panel is on the stack, no events should be
60 // passed to any panel under it. By default, all panels do this.
61 bool TrapAllEvents() const noexcept;
62 // Check if this panel can be "interrupted" to return to the main menu.
63 bool IsInterruptible() const noexcept;
64
65 // Clear the list of clickable zones.
66 void ClearZones();
67 // Add a clickable zone to the panel.
68 void AddZone(const Rectangle &rect, const std::function<void()> &fun);
69 void AddZone(const Rectangle &rect, SDL_Keycode key);
70 // Check if a click at the given coordinates triggers a clickable zone. If
71 // so, apply that zone's action and return true.

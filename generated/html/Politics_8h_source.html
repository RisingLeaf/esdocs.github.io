30// This class represents the current state of relationships between governments
31// in the game, and in particular the relationship of each government to the
32// player. The player has a reputation with each government, which is affected
33// by what they do for a government or its allies or enemies.
34class Politics {
35public:
36 // Reset to the initial political state defined in the game data.
37 void Reset();
38
39 bool IsEnemy(const Government *first, const Government *second) const;
40
41 // Commit the given "offense" against the given government (which may not
42 // actually consider it to be an offense). This may result in temporary
43 // hostilities (if the event type is PROVOKE), or a permanent change to your
44 // reputation.
45 void Offend(const Government *gov, int eventType, int count = 1);
46 // Bribe the given government to be friendly to you for one day.
47 void Bribe(const Government *gov);
48
49 // Check if the given ship can land on the given planet.
50 bool CanLand(const Ship &ship, const Planet *planet) const;
51 // Check if the player can land on the given planet.
52 bool CanLand(const Planet *planet) const;
53 bool CanUseServices(const Planet *planet) const;
54 // Bribe a planet to let the player's ships land there.
55 void BribePlanet(const Planet *planet, bool fullAccess);
56 void DominatePlanet(const Planet *planet, bool dominate = true);
57 bool HasDominated(const Planet *planet) const;
58
59 // Check to see if the player has done anything they should be fined for.
60 // Each government can only fine you once per day.
61 std::string Fine(PlayerInfo &player, const Government *gov, int scan, const Ship *target, double security);
62
63 // Get or set your reputation with the given government.
64 double Reputation(const Government *gov) const;
65 void AddReputation(const Government *gov, double value);
66 void SetReputation(const Government *gov, double value);
67
68 // Reset any temporary effects (typically because a day has passed).
69 void ResetDaily();
70
71
72private:
73 // attitude[target][other] stores how much an action toward the given target
74 // government will affect your reputation with the given other government.
75 // The relationships need not be perfectly symmetrical. For example, just
76 // because Republic ships will help a merchant under attack does not mean
77 // that merchants will come to the aid of Republic ships.
78 std::map<const Government *, double> reputationWith;
79 std::set<const Government *> provoked;
80 std::set<const Government *> bribed;
81 std::map<const Planet *, bool> bribedPlanets;
82 std::set<const Planet *> dominatedPlanets;
83 std::set<const Government *> fined;
84};

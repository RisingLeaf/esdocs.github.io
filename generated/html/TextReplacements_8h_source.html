30// A class containing a list of text replacements. Text replacements consist of a
31// key to search for and the text to replace it with. One key can have multiple potential
32// replacement texts, with the specific text chosen being defined by whichever replacement
33// is the last valid replacement for that key, with replacement validity being defined
34// by a ConditionSet.
35class TextReplacements {
36public:
37 // Load a substitutions node.
38 void Load(const DataNode &node);
39
40 // Clear this TextReplacement's substitutions and insert the substitutions of other.
41 void Revert(TextReplacements &other);
42
43 // Add new text replacements to the given map after evaluating all possible replacements.
44 // This TextReplacements will overwrite the value of any existing keys in the given map
45 // if the map and this TextReplacements share a key.
46 void Substitutions(std::map<std::string, std::string> &subs, const ConditionsStore &conditions) const;
47
48
49private:
50 // Vector with "string to be replaced", "condition when to replace", and "replacement text".
51 std::vector<std::pair<std::string, std::pair<ConditionSet, std::string>>> substitutions;
52};

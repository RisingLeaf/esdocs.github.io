26// File paths and file handling are different on each operating system. This
27// class stores the path, on each operating system, to the game's resources -
28// images, data files, etc. - and also to the "configuration" directory where
29// saved games and other user-specific information can be stored. It also
30// provides an interface for file operations so that the rest of the code can
31// be completely platform-agnostic.
32class Files {
33public:
34 static void Init(const char * const *argv);

39 static const std::string &Config();
40
41 static const std::string &Data();

43 static const std::string &Sounds();
44 static const std::string &Saves();

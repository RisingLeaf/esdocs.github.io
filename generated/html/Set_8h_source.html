24// Template representing a set of named objects of a given type, where you can
25// query it for a pointer to any object and it will return one, whether or not that
26// object has been loaded yet. (This allows cyclic pointers.)
27template<class Type>
28class Set {
29public:
30 // Allow non-const access to the owner of this set; it can hand off only
31 // const references to avoid anyone else modifying the objects.
32 Type *Get(const std::string &name) { return &data[name]; }
33 const Type *Get(const std::string &name) const { return &data[name]; }
34 // If an item already exists in this set, get it. Otherwise, return a null
35 // pointer rather than creating the item.
36 const Type *Find(const std::string &name) const;
37
38 bool Has(const std::string &name) const { return data.count(name); }
39
40 typename std::map<std::string, Type>::iterator begin() { return data.begin(); }
41 typename std::map<std::string, Type>::const_iterator begin() const { return data.begin(); }
42 typename std::map<std::string, Type>::const_iterator find(const std::string &key) const { return data.find(key); }
43 typename std::map<std::string, Type>::iterator end() { return data.end(); }
44 typename std::map<std::string, Type>::const_iterator end() const { return data.end(); }
45
46 int size() const { return data.size(); }
47 bool empty() const { return data.empty(); }
48 // Remove any objects in this set that are not in the given set, and for
49 // those that are in the given set, revert to their contents.
50 void Revert(const Set<Type> &other);
51
52
53private:
54 mutable std::map<std::string, Type> data;
55};

60const Type *Set<Type>::Find(const std::string &name) const

68template <class Type>
69void Set<Type>::Revert(const Set<Type> &other)
70{
71 auto it = data.begin();
72 auto oit = other.data.begin();
73
74 while(it != data.end())
75 {
76 if(oit == other.data.end() || it->first < oit->first)
77 it = data.erase(it);
78 else if(it->first == oit->first)
79 {
80 // If this is an entry that is in the set we are reverting to, copy
81 // the state we are reverting to.
82 it->second = oit->second;
83 ++it;
84 ++oit;
85 }
86
87 // There should never be a case when an entry in the set we are
88 // reverting to has a name that is not also in this set.
89 }
90}

32// This class represents the cargo and passengers that a ship can carry. That
33// can include ordinary commodities, plundered outfits, and mission cargo. When
34// you land on a planet, all your cargo is pooled into a single collection, not
35// tied to any one ship, so you retain it even if you sell off all your ships.
36// When you take off, cargo is distributed among your ships, and if some of it
37// will not fit it must be sold off.
38class CargoHold {
39public:
40 void Clear();
41
42 // Load the cargo manifest from a DataFile. This must be done after the
43 // GameData is loaded, so that the sizes of any outfits are known.
44 void Load(const DataNode &node);
45 // Save the cargo manifest to a file.
46 void Save(DataWriter &out) const;
47
48 // Set the capacity of this cargo hold.
49 void SetSize(int tons);
50 int Size() const;
51 int Free() const;
52 double FreePrecise() const;
53 int Used() const;
54 double UsedPrecise() const;
55 int CommoditiesSize() const;
56 int OutfitsSize() const;
57 double OutfitsSizePrecise() const;
58 bool HasOutfits() const;
59 int MissionCargoSize() const;
60 bool HasMissionCargo() const;
61 bool IsEmpty() const;
62
63 // Set the number of free bunks for passengers.
64 void SetBunks(int count);
65 int BunksFree() const;
66 int Passengers() const;
67
68 // Normal cargo:
69 int Get(const std::string &commodity) const;
70 // Spare outfits:
71 int Get(const Outfit *outfit) const;
72 // Mission cargo:
73 int Get(const Mission *mission) const;
74 int GetPassengers(const Mission *mission) const;
75
76 const std::map<std::string, int> &Commodities() const;
77 const std::map<const Outfit *, int> &Outfits() const;
78 // Note: some missions may have cargo that takes up 0 space, but should
79 // still show up on the cargo listing.
80 const std::map<const Mission *, int> &MissionCargo() const;
81 const std::map<const Mission *, int> &PassengerList() const;
82
83 // For all the transfer functions, the "other" can be null if you simply want
84 // the commodity to "disappear" or, if the "amount" is negative, to have an
85 // unlimited supply. The return value is the actual number transferred.
86 int Transfer(const std::string &commodity, int amount, CargoHold &to);
87 int Transfer(const Outfit *outfit, int amount, CargoHold &to);
88 int Transfer(const Mission *mission, int amount, CargoHold &to);
89 int TransferPassengers(const Mission *mission, int amount, CargoHold &to);
90 // Transfer as much as the given cargo hold has capacity for. The priority is
91 // first mission cargo, then spare outfits, then ordinary commodities.
92 void TransferAll(CargoHold &to, bool transferPassengers = true);
93
94 // These functions do the same thing as Transfer() with no destination
95 // specified, but they have clearer names to make the code more readable.
96 int Add(const std::string &commodity, int amount = 1);
97 int Add(const Outfit *outfit, int amount = 1);
98 int Remove(const std::string &commodity, int amount = 1);
99 int Remove(const Outfit *outfit, int amount = 1);
100
101 // Add or remove any cargo or passengers associated with the given mission.
102 void AddMissionCargo(const Mission *mission);
103 void RemoveMissionCargo(const Mission *mission);
104
105 // Get the total value of all this cargo, in the given system.
106 int64_t Value(const System *system) const;
107
108 // If anything you are carrying is illegal, return the maximum fine you can
109 // be charged for any illegal outfits plus the sum of the fines for all
110 // missions. If the returned value is negative, you are carrying something so
111 // bad that it warrants a death sentence.
112 int IllegalCargoFine(const Government *government) const;
113
114 // Returns the amount tons of illegal cargo.
115 int IllegalCargoAmount() const;
116
117
118private:
119 // Use -1 to indicate unlimited capacity.
120 int size = -1;
121 int bunks = -1;
122
123 // Track how many objects of each type are being carried:
124 std::map<std::string, int> commodities;
125 std::map<const Outfit *, int> outfits;
126 std::map<const Mission *, int> missionCargo;
127 std::map<const Mission *, int> passengers;
128};

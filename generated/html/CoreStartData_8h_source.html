31// Base class containing data of a starting scenario that is useful for later
32// reference (e.g. determining the in-game starting date, where the player began,
33// or how financially secure they were). One-time information, such as ships,
34// conditions, and the conversation, are not saved. Scenario authors desiring this
35// data should encode it into the applied starting conditions.
36class CoreStartData {
37public:
38 void Load(const DataNode &node);
39 void Save(DataWriter &out) const;
40
41 // The planet on which the player begins (or New Boston, if not set).
42 const Planet &GetPlanet() const;
43 // The system in which the game begins (or Rutilicus, if not set).
44 const System &GetSystem() const;
45 // The date on which the game begins (or 16 Nov 3013, if not set).
46 Date GetDate() const;
47 // The initial credits, debts, and credit rating for the player.
48 const Account &GetAccounts() const noexcept;
49
50 // Get the internal identifier for this starting scenario.
51 const std::string &Identifier() const noexcept;
52
53
54protected:
55 // Returns true if the child node was handled by this class.
56 bool LoadChild(const DataNode &child, bool isAdd);
57
58
59protected:
60 // The planet on which the game begins.
61 const Planet *planet = nullptr;
62 // The system in which the game begins.
63 const System *system = nullptr;
64 // The date on which the game begins.
65 Date date;
66 // Initial credits, debts, and credit rating.
67 Account accounts;
68 // The key, if any, used to identify this start in data files.
69 std::string identifier;
70};

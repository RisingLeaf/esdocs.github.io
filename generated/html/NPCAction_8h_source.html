30// A wrapper for a MissionAction that can be triggered by an NPC. Records whether
31// the action has already been done previously, which NPC checks to prevent
32// the action from being done more than once.
33// More functionality to come later, such as changing the personality of an NPC
34// on the fly.
35class NPCAction {
36public:
37 NPCAction() = default;
38 // Construct and Load() at the same time.
39 explicit NPCAction(const DataNode &node);
40
41 void Load(const DataNode &node);
42 // Note: the Save() function can assume this is an instantiated mission, not
43 // a template, so it only has to save a subset of the data.
44 void Save(DataWriter &out) const;
45 // Determine if this NPCAction references content that is not fully defined.
46 std::string Validate() const;
47
48 // Perform this action.
49 void Do(PlayerInfo &player, UI *ui = nullptr, const Mission *caller = nullptr);
50
51 // "Instantiate" this action by filling in the wildcard text for the actual
52 // destination, payment, cargo, etc.
53 NPCAction Instantiate(std::map<std::string, std::string> &subs,
54 const System *origin, int jumps, int64_t payload) const;
55
56
57private:
58 std::string trigger;

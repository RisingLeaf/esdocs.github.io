35// This class represents a set of constraints on a randomly chosen ship, planet,
36// or system. For example, it can require that the planet used for a mission
37// have a certain attribute or be owned by a certain government, or be a
38// certain distance away from the current system.
39class LocationFilter {
40public:
41 LocationFilter() noexcept = default;
42 // Construct and Load() at the same time.
43 explicit LocationFilter(const DataNode &node);
44
45 // Examine all the children of the given node and load any that are filters.
46 void Load(const DataNode &node);
47 // This only saves the children. Save the root node separately. It does
48 // handle indenting, however.
49 void Save(DataWriter &out) const;
50
51 // Check if this filter contains any specifications.
52 bool IsEmpty() const;
53 bool IsValid() const;
54
55 // If the player is in the given system, does this filter match?
56 bool Matches(const Planet *planet, const System *origin = nullptr) const;
57 bool Matches(const System *system, const System *origin = nullptr) const;
58 // Ships are chosen based on system/"near" filters, government, category
59 // of ship, outfits installed/carried, and their total attributes.
60 bool Matches(const Ship &ship) const;
61
62 // Return a new LocationFilter with any "distance" conditions converted
63 // into "near" references, relative to the given system.
64 LocationFilter SetOrigin(const System *origin) const;
65 // Generic find system / find planet methods, based on the given origin
66 // system (e.g. the player's current system) and ability to land.
67 const System *PickSystem(const System *origin) const;
68 const Planet *PickPlanet(const System *origin, bool hasClearance = false, bool requireSpaceport = true) const;
69
70
71private:
72 // Load one particular line of conditions.
73 void LoadChild(const DataNode &child);
74 // Check if the filter matches the given system. If it did not, return true
75 // only if the filter wasn't looking for planet characteristics or if the
76 // didPlanet argument is set (meaning we already checked those).
77 bool Matches(const System *system, const System *origin, bool didPlanet) const;
78
79
80private:
81 bool isEmpty = true;
82
83 // The planet must satisfy these conditions:
84 std::set<const Planet *> planets;
85 // It must have at least one attribute from each set in this list:
86 std::list<std::set<std::string>> attributes;
87
88 // The system must satisfy these conditions:
89 std::set<const System *> systems;
90 std::set<const Government *> governments;
91 // The reference point and distance limits of a "near <system>" filter.
92 const System *center = nullptr;
93 int centerMinDistance = 0;
94 int centerMaxDistance = 1;
95 DistanceCalculationSettings centerDistanceOptions;
96 // Distance limits used in a "distance" filter.
97 int originMinDistance = 0;
98 int originMaxDistance = -1;
99 DistanceCalculationSettings originDistanceOptions;
100
101 // At least one of the outfits from each set must be available
102 // (to purchase or plunder):
103 std::list<std::set<const Outfit *>> outfits;
104 // A ship must belong to one of these categories:
105 std::set<std::string> shipCategory;
106
107 // These filters store all the things the planet, system, or ship must not be.
108 std::list<LocationFilter> notFilters;
109 // These filters store all the things the planet or system must border.
110 std::list<LocationFilter> neighborFilters;
111};

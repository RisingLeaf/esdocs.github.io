29// Class representing a port on a planet and its capabilities, such as what properties
30// of a ship it can recharge and what services (e.g. banking, trading) it can provide.
31class Port
32{
33public:
34 // The different ship properties that can be recharged by a port.
35 class RechargeType
36 {
37 public:
38 static constexpr int None = 0;
39 static constexpr int Shields = (1 << 0);
40 static constexpr int Hull = (1 << 1);
41 static constexpr int Energy = (1 << 2);
42 static constexpr int Fuel = (1 << 3);
43 static constexpr int All = Shields | Hull | Energy | Fuel;
44 };
45
46 // The different services available on this port.
47 class ServicesType
48 {
49 public:
50 static constexpr int None = 0;
51 static constexpr int Trading = (1 << 0);
52 static constexpr int JobBoard = (1 << 1);
53 static constexpr int Bank = (1 << 2);
54 static constexpr int HireCrew = (1 << 3);
55 static constexpr int OffersMissions = (1 << 4);
56 static constexpr int All = Trading | JobBoard | Bank | HireCrew | OffersMissions;
57 };
58
59
60public:
61 // Load a port's description from a node.
62 void Load(const DataNode &node);
63 void LoadDefaultSpaceport();
64 void LoadUninhabitedSpaceport();
65
66 // Whether this port was loaded from the Load function.
67 bool CustomLoaded() const;
68
69 // Whether this port has any services available.
70 bool HasServices() const;
71
72 // Get all the possible sources that can get recharged at this port.
73 int GetRecharges() const;
74
75 const std::string &Name() const;

77 const std::string &Description() const;
78
79 // Check whether the given recharging is possible.

82 bool HasService(int type) const;
83
84 bool HasNews() const;
85
86
87private:

90

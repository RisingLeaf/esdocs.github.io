32// Class wrapping IETF v4 GUIDs, providing lazy initialization.
33class EsUuid final {
34public:
35 // Used to represent a UUID across supported platforms.
36 struct UuidType final {
37 UuidType() = default;
38 UuidType(UuidType &&) = default;
39 UuidType &operator=(UuidType &&) = default;
40 ~UuidType() = default;
41 UuidType &operator=(const UuidType &other) { return *this = UuidType(other); }
42#if defined(_WIN32)
43 UuidType(const UuidType &other) = default;
44 UUID id = {};
45#else
46 UuidType(const UuidType &other) { uuid_copy(id, other.id); }
47 uuid_t id = {};
48#endif
49 };

51
52public:
53 static EsUuid FromString(const std::string &input);
54 EsUuid() noexcept = default;
55 ~EsUuid() noexcept = default;
56 // Copying a UUID does not copy its value. (This allows us to use simple copy operations on stock
57 // ship definitions when spawning fleets, etc.)
58 EsUuid(const EsUuid &other) noexcept : value{} {};
59 // Copy-assigning also results in an empty UUID.
60 EsUuid &operator=(const EsUuid &other) noexcept { return *this = EsUuid(other); };
61 // UUIDs can be move-constructed as-is.
62 EsUuid(EsUuid &&) noexcept = default;
63 // UUIDs can be move-assigned as-is.
64 EsUuid &operator=(EsUuid &&) noexcept = default;

66 // UUIDs can be compared against other UUIDs.

68 bool operator!=(const EsUuid &other) const noexcept(false);
69 bool operator<(const EsUuid &other) const noexcept(false);

85private:

91template <class T>
92struct UUIDComparator {
93 // Comparator for collections of shared_ptr<T>
94 bool operator() (const std::shared_ptr<T> &a, const std::shared_ptr<T> &b) const noexcept(false)
95 {
96 return a->UUID() < b->UUID();
97 }
98
99 // Comparator for collections of T*, e.g. set<T *>
100 bool operator()(const T *a, const T *b) const noexcept(false)
101 {
102 return a->UUID() < b->UUID();
103 }
104 // No comparator for collections of T, as std containers generally perform copy operations
105 // and copying this class will eventually be disabled.
106};

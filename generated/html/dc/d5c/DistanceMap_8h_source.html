32// This is a map of how many hyperspace jumps it takes to get to other systems
33// from the given "center" system. Ships with a hyperdrive travel using the
34// "links" between systems. Ships with jump drives can make use of those links,
35// but can also travel to any of a system's "neighbors." A distance map can also
36// be used to calculate the shortest route between two systems.
37class DistanceMap {
38public:
39 // Find paths to the given system. The optional arguments put a limit on how
40 // many systems will be returned and how far away they are allowed to be.
41 explicit DistanceMap(const System *center, int maxCount = -1, int maxDistance = -1);
42 // Find paths to the given system, potentially using wormholes, a jump drive, or both.
43 // Optional arguments are as above.
44 explicit DistanceMap(const System *center, WormholeStrategy wormholeStrategy,
45 bool useJumpDrive, int maxCount = -1, int maxDistance = -1);
46 // If a player is given, the map will only use hyperspace paths known to the
47 // player; that is, one end of the path has been visited. Also, if the
48 // player's flagship has a jump drive, the jumps will be make use of it.
49 explicit DistanceMap(const PlayerInfo &player, const System *center = nullptr);
50 // Calculate the path for the given ship to get to the given system. The
51 // ship will use a jump drive or hyperdrive depending on what it has. The
52 // pathfinding will stop once a path to the destination is found.
53 // If a player is given, the path will only include systems that the
54 // player has visited.
55 DistanceMap(const Ship &ship, const System *destination, const PlayerInfo *player = nullptr);
56
57 // Find out if the given system is reachable.
58 bool HasRoute(const System *system) const;
59 // Find out how many days away the given system is.
60 int Days(const System *system) const;
61 // Starting in the given system, what is the next system along the route?
62 const System *Route(const System *system) const;
63
64 // Get a set containing all the systems.
65 std::set<const System *> Systems() const;
66 // Get the end of the route.
67 const System *End() const;
68
69 // How much fuel is needed to travel between two systems.
70 int RequiredFuel(const System *system1, const System *system2) const;
71
72
73private:
74 // For each system, track how much fuel it will take to get there, how many
75 // days, how much danger you will pass through, and where you will go next.
76 class Edge {
77 public:
78 explicit Edge(const System *system = nullptr);
79
80 // Sorting operator to prioritize the "best" edges. The priority queue
81 // returns the "largest" item, so this should return true if this item
82 // is lower priority than the given item.
83 bool operator<(const Edge &other) const;
84
85 const System *next = nullptr;
86 int fuel = 0;
87 int days = 0;
88 double danger = 0.;
89 };
90
91
92private:
93 // Depending on the capabilities of the given ship, use hyperspace paths,
94 // jump drive paths, or both to find the shortest route. Bail out if the
95 // source system or the maximum count is reached.
96 void Init(const Ship *ship = nullptr);
97 // Add the given links to the map. Return false if an end condition is hit.
98 bool Propagate(Edge edge, bool useJump);
99 // Check if we already have a better path to the given system.
100 bool HasBetter(const System &to, const Edge &edge);
101 // Add the given path to the record.
102 void Add(const System &to, Edge edge);
103 // Check whether the given link is travelable. If no player was given in the
104 // constructor then this is always true; otherwise, the player must know
105 // that the given link exists.
106 bool CheckLink(const System &from, const System &to, bool useJump) const;
107
108
109private:
110 std::map<const System *, Edge> route;
111
112 // Variables only used during construction:
113 std::priority_queue<Edge> edges;
114 const PlayerInfo *player = nullptr;
115 const System *source = nullptr;
116 const System *center = nullptr;
117 WormholeStrategy wormholeStrategy = WormholeStrategy::ALL;
118 int maxCount = -1;
119 int maxDistance = -1;
120 // How much fuel is used for travel. If either value is zero, it means that
121 // the ship does not have that type of drive.
122 int hyperspaceFuel = 100;
123 int jumpFuel = 0;
124 double jumpRange = 0.;
125 const Ship *ship = nullptr;
126};

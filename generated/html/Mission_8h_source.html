45// This class represents a mission that the player can take on, such as carrying
46// cargo from one planet to another or defending or attacking a ship. Many of
47// a mission's parameters can be randomized, either so that the same mission
48// template can be reused many times, or just so the mission is not always
49// exactly the same every time you replay the game.
50class Mission {
51public:
52 Mission() = default;
53 // Copying a mission instance isn't allowed.
54 Mission(const Mission &) = delete;
55 Mission &operator=(const Mission &) = delete;
56 Mission(Mission &&) = default;
57 Mission &operator=(Mission &&) = default;
58 ~Mission() noexcept = default;
59
60 // Construct and Load() at the same time.
61 explicit Mission(const DataNode &node);
62
63 // Load a mission, either from the game data or from a saved game.
64 void Load(const DataNode &node);
65 // Save a mission. It is safe to assume that any mission that is being saved
66 // is already "instantiated," so only a subset of the data must be saved.
67 void Save(DataWriter &out, const std::string &tag = "mission") const;
68 // Add "never" to the toOffer ConditionSet, preventing this mission from offering.
69 void NeverOffer();
70
71 // Basic mission information.
72 const EsUuid &UUID() const noexcept;
73 const std::string &Name() const;
74 const std::string &Description() const;
75 // Check if this mission should be shown in your mission list. If not, the
76 // player will not know this mission exists (which is sometimes useful).
77 bool IsVisible() const;
78 // Check if this mission should be quarantined due to requiring currently-
79 // undefined ships, planets, or systems (i.e. is from an inactive plugin).
80 bool IsValid() const;
81 // Check if this mission has high priority. If any high-priority missions
82 // are available, no others will be shown at landing or in the spaceport.
83 // This is to be used for missions that are part of a series.
84 bool HasPriority() const;
85 // Check if this mission is a "minor" mission. Minor missions will only be
86 // offered if no other missions (minor or otherwise) are being offered.
87 bool IsMinor() const;
88
89 // Find out where this mission is offered.
90 enum Location {SPACEPORT, LANDING, JOB, ASSISTING, BOARDING, SHIPYARD, OUTFITTER};
91 bool IsAtLocation(Location location) const;
92
93 // Information about what you are doing.
94 const Ship *SourceShip() const;
95 const Planet *Destination() const;
96 const std::set<const System *> &Waypoints() const;
97 const std::set<const System *> &VisitedWaypoints() const;
98 const std::set<const Planet *> &Stopovers() const;
99 const std::set<const Planet *> &VisitedStopovers() const;
100 const std::string &Cargo() const;
101 int CargoSize() const;
102 int IllegalCargoFine() const;
103 std::string IllegalCargoMessage() const;
104 bool FailIfDiscovered() const;
105 int Passengers() const;
106 int64_t DisplayedPayment() const;
107 // The mission should take this many jumps.
108 // Only matters to available jobs (not saved to file)
109 const int ExpectedJumps() const;
110 int CalculateJumps(const System *const sourceSystem);
111 // The mission must be completed by this deadline (if there is a deadline).
112 const Date &Deadline() const;
113 // If this mission's deadline was before the given date and it has not been
114 // marked as failing already, mark it and return true.
115 bool CheckDeadline(const Date &today);
116 // Check if you have special clearance to land on your destination.
117 bool HasClearance(const Planet *planet) const;
118 // Get the string to be shown in the destination planet's hailing dialog. If
119 // this is "auto", you don't have to hail them to get landing permission.
120 const std::string &ClearanceMessage() const;
121 // Check whether we have full clearance to land and use the planet's
122 // services, or whether we are landing in secret ("infiltrating").
123 bool HasFullClearance() const;
124
125 // Check if it's possible to offer or complete this mission right now. The
126 // check for whether you can offer a mission does not take available space
127 // into account, so before actually offering a mission you should also check
128 // if the player has enough space.
129 bool CanOffer(const PlayerInfo &player, const std::shared_ptr<Ship> &boardingShip = nullptr) const;
130 bool CanAccept(const PlayerInfo &player) const;
131 bool HasSpace(const PlayerInfo &player) const;
132 bool HasSpace(const Ship &ship) const;
133 bool CanComplete(const PlayerInfo &player) const;
134 bool IsSatisfied(const PlayerInfo &player) const;
135 bool HasFailed(const PlayerInfo &player) const;
136 bool IsFailed() const;
137 bool OverridesCapture() const;
138 // Mark a mission failed (e.g. due to a "fail" action in another mission).
139 void Fail();
140 // Get a string to show if this mission is "blocked" from being offered
141 // because it requires you to have more passenger or cargo space free. After
142 // calling this function, any future calls to it will return an empty string
143 // so that you do not display the same message multiple times.
144 std::string BlockedMessage(const PlayerInfo &player);
145 // Check if this mission recommends that the game be autosaved when it is
146 // accepted. This should be set for main story line missions that have a
147 // high chance of failing, such as escort missions.
148 bool RecommendsAutosave() const;
149 // Check if this mission is unique, i.e. not something that will be offered
150 // over and over again in different variants.
151 bool IsUnique() const;
152
153 // When the state of this mission changes, it may make changes to the player
154 // information or show new UI panels. PlayerInfo::MissionCallback() will be
155 // used as the callback for an `on offer` conversation, to handle its response.
156 // If it is not possible for this change to happen, this function returns false.
157 enum Trigger {COMPLETE, OFFER, ACCEPT, DECLINE, FAIL, ABORT, DEFER, VISIT, STOPOVER, WAYPOINT, DAILY, DISABLED};
158 bool Do(Trigger trigger, PlayerInfo &player, UI *ui = nullptr, const std::shared_ptr<Ship> &boardingShip = nullptr);
159
160 // Get a list of NPCs associated with this mission. Every time the player
161 // takes off from a planet, they should be added to the active ships.
162 const std::list<NPC> &NPCs() const;
163 // Update which NPCs are active based on their spawn and despawn conditions.
164 void UpdateNPCs(const PlayerInfo &player);
165 // Checks if the given ship belongs to one of the mission's NPCs.
166 bool HasShip(const std::shared_ptr<Ship> &ship) const;
167 // If any event occurs between two ships, check to see if this mission cares
168 // about it. This may affect the mission status or display a message.
169 void Do(const ShipEvent &event, PlayerInfo &player, UI *ui);
170 bool RequiresGiftedShip(const std::string &shipId) const;
171
172 // Get the internal name used for this mission. This name is unique and is
173 // never modified by string substitution, so it can be used in condition
174 // variables, etc.
175 const std::string &Identifier() const;
176 // Get a specific mission action from this mission.
177 // If the mission action is not found for the given trigger, returns an empty
178 // mission action.
179 const MissionAction &GetAction(Trigger trigger) const;
180
181 // "Instantiate" a mission by replacing randomly selected values and places
182 // with a single choice, and then replacing any wildcard text as well.
183 Mission Instantiate(const PlayerInfo &player, const std::shared_ptr<Ship> &boardingShip = nullptr) const;
184

187 bool Enter(const System *system, PlayerInfo &player, UI *ui);

265 std::set<const MissionAction *> didEnter;

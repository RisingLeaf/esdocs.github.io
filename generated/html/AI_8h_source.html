44// This class is responsible for controlling all the ships in the game,
45// including the player's "flagship" - which is usually controlled via the
46// keyboard but can also make use of the AI's autopilot for landing, aiming,
47// or hyperspace travel between systems. The AI also tracks which actions
48// ships have performed, to avoid having the same ship try to scan or board
49// the same target over and over.
50class AI {
51public:
52 // Any object that can be a ship's target is in a list of this type:
53template <class Type>
54 using List = std::list<std::shared_ptr<Type>>;
55 // Constructor, giving the AI access to the player and various object lists.
56 AI(const PlayerInfo &player, const List<Ship> &ships,
57 const List<Minable> &minables, const List<Flotsam> &flotsam);
58
59 // Fleet commands from the player.
60 void IssueShipTarget(const std::shared_ptr<Ship> &target);
61 void IssueAsteroidTarget(const std::shared_ptr<Minable> &targetAsteroid);
62 void IssueMoveTarget(const Point &target, const System *moveToSystem);
63 // Commands issued via the keyboard (mostly, to the flagship).
64 void UpdateKeys(PlayerInfo &player, Command &clickCommands);
65
66 // Allow the AI to track any events it is interested in.
67 void UpdateEvents(const std::list<ShipEvent> &events);
68 // Reset the AI's memory of events.
69 void Clean();
70 // Clear ship orders. This should be done when the player lands on a planet,
71 // but not when they jump from one system to another.
72 void ClearOrders();
73 // Issue AI commands to all ships for one game step.
74 void Step(Command &activeCommands);
75
76 // Set the mouse position for turning the player's flagship.
77 void SetMousePosition(Point position);
78
79 // Get the in-system strength of each government's allies and enemies.
80 int64_t AllyStrength(const Government *government);

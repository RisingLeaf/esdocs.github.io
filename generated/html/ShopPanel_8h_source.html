52 // BuyResult holds the result of an attempt to buy. It is implicitly
53 // created from a string or boolean in code. Any string indicates failure.
54 // True indicates success, of course, while false (without a string)
55 // indicates failure, but no need to pop up a message about it.
56 class BuyResult {
57 public:
58 BuyResult(const char *error) : success(false), message(error) {}
59 BuyResult(std::string error) : success(false), message(std::move(error)) {}
60 BuyResult(bool result) : success(result), message() {}
61
62 explicit operator bool() const noexcept { return success; }
63
64 bool HasMessage() const noexcept { return !message.empty(); }
65 const std::string &Message() const noexcept { return message; }
66
67
68 private:
69 bool success = true;
70 std::string message;
71 };

25// This class stores the odds that one ship will be able to capture another, and
26// can report the odds for any number of crew up to what each ship starts out
27// with (because the odds change each time a crew member is lost). If either
28// ship has hand-to-hand weapons available, the crew members will make use of
29// them starting with whatever weapon is most useful to them. In each round of
30// combat, one ship will lose one crew member. Which ship loses depends on the
31// ratio of the strengths of the two crews (plus weapons), and whether each crew
32// is attacking or defending; defending crew get a +1 power bonus.
33class CaptureOdds {
34public:
35 // Calculate odds that the first given ship can capture the second, assuming
36 // the first ship always attacks and the second one always defends.
37 CaptureOdds(const Ship &attacker, const Ship &defender);
38
39 // Get the odds of the attacker winning if the two ships have the given
40 // number of crew members remaining.
41 double Odds(int attackingCrew, int defendingCrew) const;
42 // Get the expected number of casualties in the remainder of the battle if
43 // the two ships have the given number of crew remaining.
44 double AttackerCasualties(int attackingCrew, int defendingCrew) const;
45 double DefenderCasualties(int attackingCrew, int defendingCrew) const;
46
47 // Get the total power (inherent crew power plus bonuses from hand to hand
48 // weapons) for each ship when they have the given number of crew remaining.
49 double AttackerPower(int attackingCrew) const;
50 double DefenderPower(int defendingCrew) const;
51
52
53private:
54 // Generate the lookup table.
55 void Calculate();
56 // Map crew numbers into an index in the lookup table.
57 int Index(int attackingCrew, int defendingCrew) const;
58
59 // Calculate attack or defense power for each number of crew members up to
60 // the given ship's full complement.
61 static std::vector<double> Power(const Ship &ship, bool isDefender);
62
63
64private:
65 // Attacker and defender power lookup tables.
66 std::vector<double> powerA;
67 std::vector<double> powerD;
68
69 // Capture odds lookup table.
70 std::vector<double> capture;
71 // Expected casualties lookup table.
72 std::vector<double> casualtiesA;
73 std::vector<double> casualtiesD;
74};

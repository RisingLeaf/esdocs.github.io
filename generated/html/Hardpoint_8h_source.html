33// A single weapon hardpoint on the ship (i.e. a gun port or turret mount),
34// which may or may not have a weapon installed.
35class Hardpoint {
36public:
37 // Constructor. Hardpoints may or may not specify what weapon is in them.
38 Hardpoint(const Point &point, const Angle &baseAngle, bool isTurret,
39 bool isParallel, bool isUnder, const Outfit *outfit = nullptr);
40
41 // Get the weapon installed in this hardpoint (or null if there is none).
42 const Outfit *GetOutfit() const;
43 // Get the location, relative to the center of the ship, from which
44 // projectiles of this weapon should originate. This point must be
45 // rotated to take the ship's current facing direction into account.
46 const Point &GetPoint() const;
47 // Get the angle that this weapon is aimed at, relative to the ship.
48 const Angle &GetAngle() const;
49 // Get the base angle that this weapon is aimed at (without harmonization/convergence), relative to the ship.
50 const Angle &GetBaseAngle() const;
51 // Get the angle this weapon ought to point at for ideal gun harmonization.
52 Angle HarmonizedAngle() const;
53 // Shortcuts for querying weapon characteristics.
54 bool IsTurret() const;
55 bool IsParallel() const;
56 bool IsUnder() const;
57 bool IsHoming() const;
58 bool IsSpecial() const;
59 bool CanAim() const;
60
61 // Check if this weapon is ready to fire.
62 bool IsReady() const;
63 // Check if this weapon was firing in the previous step.
64 bool WasFiring() const;
65 // If this is a burst weapon, get the number of shots left in the burst.
66 int BurstRemaining() const;
67 // Perform one step (i.e. decrement the reload count).
68 void Step();
69
70 // Adjust this weapon's aim by the given amount, relative to its maximum
71 // "turret turn" rate.
72 void Aim(double amount);
73 // Fire this weapon. If it is a turret, it automatically points toward
74 // the given ship's target. If the weapon requires ammunition, it will
75 // be subtracted from the given ship.
76 void Fire(Ship &ship, std::vector<Projectile> &projectiles, std::vector<Visual> &visuals);
77 // Fire an anti-missile. Returns true if the missile should be killed.
78 bool FireAntiMissile(Ship &ship, const Projectile &projectile, std::vector<Visual> &visuals);
79 // Fire a tractor beam. Returns true if the flotsam was hit.
80 bool FireTractorBeam(Ship &ship, const Flotsam &flotsam, std::vector<Visual> &visuals);
81 // This weapon jammed. Increase its reload counters, but don't fire.
82 void Jam();
83
84 // Install a weapon here (assuming it is empty). This is only for
85 // Armament to call internally.
86 void Install(const Outfit *outfit);
87 // Reload this weapon.
88 void Reload();
89 // Uninstall the outfit from this port (if it has one).
90 void Uninstall();
91
92
93private:
94 // Check whether a projectile or flotsam is within the range of the anti-missile
95 // or tractor beam system and create visuals if it is.
96 bool FireSpecialSystem(Ship &ship, const Body &body, std::vector<Visual> &visuals);
97 // Reset the reload counters and expend ammunition, if any.
98 void Fire(Ship &ship, const Point &start, const Angle &aim);
99
100
101private:
102 // The weapon installed in this hardpoint.
103 const Outfit *outfit = nullptr;
104 // Hardpoint location, in world coordinates relative to the ship's center.
105 Point point;
106 // Angle of firing direction (guns only).
107 Angle baseAngle;
108 // This hardpoint is for a turret or a gun.
109 bool isTurret = false;
110 // Indicates if this hardpoint disallows converging (guns only).
111 bool isParallel = false;
112 // Indicates whether the hardpoint sprite is drawn under the ship.
113 bool isUnder = false;
114
115 // Angle adjustment for convergence.
116 Angle angle;
117 // Reload timers and other attributes.
118 double reload = 0.;
119 double burstReload = 0.;
120 int burstCount = 0;
121 bool isFiring = false;
122 bool wasFiring = false;
123};

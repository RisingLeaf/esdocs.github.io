39// A MissionAction represents what happens when a Mission reaches a certain
40// milestone, including offered, accepted, declined, completed, or failed.
41// In addition to performing a GameAction, a MissionAction can gate the task on
42// the ownership of specific outfits and also display dialogs or conversations.
43class MissionAction {
44public:
45 MissionAction() = default;
46 // Construct and Load() at the same time.
47 MissionAction(const DataNode &node);
48
49 void Load(const DataNode &node);
50 void LoadSingle(const DataNode &node);
51 // Note: the Save() function can assume this is an instantiated mission, not
52 // a template, so it only has to save a subset of the data.
53 void Save(DataWriter &out) const;
54 void SaveBody(DataWriter &out) const;
55 // Determine if this MissionAction references content that is not fully defined.
56 std::string Validate() const;
57
58 const std::string &DialogText() const;
59
60 // Check if this action can be completed right now. It cannot be completed
61 // if it takes away money or outfits that the player does not have, or should
62 // take place in a system that does not match the specified LocationFilter.
63 bool CanBeDone(const PlayerInfo &player, const std::shared_ptr<Ship> &boardingShip = nullptr) const;
64 // Check if this action requires this ship to exist in order to ever be completed.
65 bool RequiresGiftedShip(const std::string &shipId) const;
66 // Perform this action. If a conversation is shown, the given destination
67 // will be highlighted in the map if you bring it up.
68 void Do(PlayerInfo &player, UI *ui, const Mission *caller,
69 const System *destination = nullptr, const std::shared_ptr<Ship> &ship = nullptr,
70 const bool isUnique = true) const;
71
72 // "Instantiate" this action by filling in the wildcard text for the actual
73 // destination, payment, cargo, etc.
74 MissionAction Instantiate(std::map<std::string, std::string> &subs,
75 const System *origin, int jumps, int64_t payload) const;
76
77 int64_t Payment() const noexcept;
78
79private:
80 std::string trigger;
81 std::string system;
82 LocationFilter systemFilter;
83
84 std::string dialogText;
85 ExclusiveItem<Phrase> dialogPhrase;
86 ExclusiveItem<Conversation> conversation;
87
88 // Outfits that are required to be owned (or not) for this action to be performable.
89 std::map<const Outfit *, int> requiredOutfits;
90
91 // Tasks this mission action performs, such as modifying accounts, inventory, or conditions.
92 GameAction action;
93};

32// Helper class for drawing text formatted in a table, where each column of the
33// table is aligned left, right, or centered. This also handles spacing in
34// between table rows, underlines, selection highlights, etc.
35class Table {
36public:
37 Table();
38
39 // Set the column positions. If no columns are set, the Table will draw a
40 // list (one column of text, left aligned).
41 void Clear();
42 void AddColumn(int x, Layout layout = {});
43
44 // Set the font size. Default is 14 pixels.
45 void SetFontSize(int size);
46 // Set the row height. Default is 20 pixels.
47 void SetRowHeight(int height) noexcept;
48 // Set the width of the highlight area. If the underline has not been set,
49 // this will also set the width of the underline.
50 void SetHighlight(int startX, int endX) noexcept;
51 // Set the X range of the underline. If the highlight has not been set, this
52 // will also set the width of the highlight.
53 void SetUnderline(int startX, int endX) noexcept;
54
55 // Begin drawing at the given position. Each time text is drawn, it fills a
56 // new column until all columns have been filled. Then, the Y position is
57 // increased based on the row height, and a new row begins.
58 void DrawAt(const Point &point) const;
59
60 // Set the color for drawing text and underlines.
61 void SetColor(const Color &color) const;
62
63 // Advance to the next field without drawing anything.
64 void Advance(int fields = 1) const;
65
66 // Draw a single text field, and move on to the next one.
67 void Draw(const char *text) const;
68 void Draw(const std::string &text) const;
69 // If a color is given, this field is drawn using that color, but the
70 // previously set color will be used for future fields.
71 void Draw(const char *text, const Color &color) const;
72 void Draw(const std::string &text, const Color &color) const;
73 void Draw(double value) const;
74 void Draw(double value, const Color &color) const;
75 // Use the width & alignment associated with the text (instead of the column's).
76 void DrawCustom(const DisplayText &text) const;
77 void DrawCustom(const DisplayText &text, const Color &color) const;
78
79 // Draw two columns as a pair with opposite alignments. If needed, truncate the given
80 // column based on the width of the non-truncated column's value.
81 void DrawTruncatedPair(const std::string &left, const Color &leftColor, const std::string &right,
82 const Color &rightColor, Truncate strategy, bool truncateRightColumn) const;
83
84 // Draw an underline under the text for the current row.
85 void DrawUnderline() const;
86 void DrawUnderline(const Color &color) const;
87
88 // Highlight the current row.
89 void DrawHighlight() const;
90 void DrawHighlight(const Color &color) const;
91
92 // Shift the draw position down by the given amount. This usually should not
93 // be called in the middle of a row, or the fields will not line up.
94 void DrawGap(int y) const;
95
96 // Get the point that should be passed to DrawAt() to start the next row at
97 // the given location.
98 Point GetPoint() const;
99
100 // Get the center and size of the current row. This can be used to define
101 // what screen region constitutes a mouse click on this particular row.
102 Point GetCenterPoint() const;
103 Point GetRowSize() const;
104 Rectangle GetRowBounds() const;
105
106
107private:
108 // Starting position for drawing a column is:
109 // point + Point(offset + align * font.Width(text), 0.)
110 // So, "align" will either be 0, -.5, or -1.
111 class Column {
112 public:
113 Column(double offset, Layout layout) noexcept;
114
115 double offset = 0.;
116 Layout layout;
117 };
118
119
120private:
121 void Draw(const std::string &text, const Layout *special, const Color &color) const;
122
123
124private:
125 // The current draw position.
126 mutable Point point;
127 // The column being drawn.
128 mutable std::vector<Column>::const_iterator it;
129 // The color to use on the next draw call.
130 mutable Color color;
131
132 const Font *font = nullptr;
133 Point rowSize;
134 Point center;
135 Point lineSize;
136 Point lineOff;
137
138 std::vector<Column> columns;
139};

21// Class representing an RGBA color (for use by OpenGL). The specified colors
22// should be in premultiplied alpha mode. For example, a white color which is
23// 50% translucent would be {.5, .5, .5} with an alpha of .5. This allows you to
24// also specify colors that use additive blending. For example, if the alpha is
25// zero the color's components will be added to whatever is underneath them.
26class Color {
27public:
28 // Constructor for shades of gray, opaque unless an alpha is also given.
29 explicit Color(float i = 1.f, float a = 1.f);
30 // Constructor for colors, opaque unless an alpha is also given.
31 Color(float r, float g, float b, float a = 1.f);
32
33 bool operator==(const Color &other) const;
34 bool operator!=(const Color &other) const;
35
36 // Set this color to the given RGBA values.
37 void Load(double r, double g, double b, double a);
38 // Check if Load() has been called for this color.
39 bool IsLoaded() const;
40 // Get the color as a float vector, suitable for use by OpenGL.
41 const float *Get() const;
42
43 // Get this color, but entirely opaque. That is, this is the color you would
44 // get if drawing this color on top of opaque black.
45 Color Opaque() const;
46 // Assuming that this color is completely opaque, get the same color with
47 // the given transparency.
48 Color Transparent(float alpha) const;
49 // Assuming that this is an opaque color, get its additive equivalent with
50 // the given fraction of its full brightness.
51 Color Additive(float alpha) const;
52
53 // Compute a linear combination
54 static Color Combine(float a1, Color c1, float a2, Color c2);
55
56 // Multiply the RGBA values of the given Color by the given scalar and return a new Color.
57 static Color Multiply(float scalar, const Color &base);
58
59
60private:
61 // Store the color as a float vector for easy interfacing with OpenGL.
62 float color[4];
63
64 bool isLoaded = false;
65};

23// Class representing a date (day, month, and year). Built-in date structures
24// combined dates with time within the day, which can lead to accumulated errors
25// when trying to increment dates in whole-day steps.
26class Date {
27public:
28 Date() = default;
29 Date(int day, int month, int year);
30
31 // Get this date as a string, in the form "Day, DD Mon Year".
32 const std::string &ToString() const;

34 // conversation text.
35 std::string LongString() const;
36
37 // Check if this date has been initialized.
38 explicit operator bool() const;
39 bool operator!() const;
40
41 // Move the date forward one day.
42 Date &operator++();
43 Date operator++(int);

45 Date operator+(int days) const;
46 // Get the number of days between two dates.
47 int operator-(const Date &other) const;
48 // Comparison operators.
49 bool operator<(const Date &other) const;
50 bool operator<=(const Date &other) const;
51 bool operator>(const Date &other) const;
52 bool operator>=(const Date &other) const;
53 bool operator==(const Date &other) const;
54 bool operator!=(const Date &other) const;
55
56 // Get the number of days that have elapsed since the "epoch" and the start of this year.
57 int DaysSinceEpoch() const;

60
61 // Get the date as numbers.
62 int Day() const;
63 int Month() const;
64 int Year() const;
65
66
67private:

74};

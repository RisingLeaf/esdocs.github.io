28// Class representing the outline of an object, with functions for checking if a
29// line segment intersects that object or if a point is within a certain distance.
30// The outline is represented in polygonal form, which allows intersection tests
31// to be done much more efficiently than if we were testing individual pixels in
32// the image itself.
33class Mask {
34public:
35 // Construct a mask from the alpha channel of an RGBA-formatted image.
36 void Create(const ImageBuffer &image, int frame = 0);
37
38 // Check whether a mask was successfully generated from the image.
39 bool IsLoaded() const;
40
41 // Check if this mask intersects the given line segment (from sA to vA). If
42 // it does, return the fraction of the way along the segment where the
43 // intersection occurs. The sA should be relative to this object's center,
44 // while vA should be relative to sA.
45 // If this object contains the given point, the return value is 0. If there
46 // is no collision, the return value is 1.
47 double Collide(Point sA, Point vA, Angle facing) const;
48
49 // Check whether the mask contains the given point.
50 bool Contains(Point point, Angle facing) const;
51
52 // Find out whether this object (rotated and scaled as represented by the given
53 // unit vector) is touching a ring defined by the given inner and outer ranges.
54 bool WithinRing(Point point, Angle facing, double inner, double outer) const;
55
56 // Find out how close the given point is to the mask.
57 double Range(Point point, Angle facing) const;
58 // Get the maximum distance from the center of this mask.
59 double Radius() const;
60
61 // Get the individual outlines that comprise this mask.
62 const std::vector<std::vector<Point>> &Outlines() const;
63
64 // Scale all the points in the mask.
65 Mask operator*(double scale) const;
66 friend Mask operator*(double scale, const Mask &mask);
67
68
69private:
70 double Intersection(Point sA, Point vA) const;
71 bool Contains(Point point) const;
72
73
74private:
75 std::vector<std::vector<Point>> outlines;
76 double radius = 0.;
77};

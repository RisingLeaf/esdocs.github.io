31// A condition set is a collection of operations on the player's set of named
32// "conditions". This includes "test" operations that just check the values of
33// those conditions, and other operations that can be "applied" to change the
34// values.
35class ConditionSet {
36public:
37 ConditionSet() = default;
38
39 // Construct and Load() at the same time.
40 explicit ConditionSet(const DataNode &node);
41
42 // Load a set of conditions from the children of this node. Prints a
43 // warning if an and/or node contains assignment expressions.
44 void Load(const DataNode &node);
45 // Save a set of conditions.
46 void Save(DataWriter &out) const;
47
48 // Check if there are any entries in this set.
49 bool IsEmpty() const;
50
51 // Read a single condition from a data node.
52 void Add(const DataNode &node);
53 bool Add(const std::string &firstToken, const std::string &secondToken);
54 // Add simple conditions having only a single operator.
55 bool Add(const std::string &name, const std::string &op, const std::string &value);
56 // Add complex conditions having multiple operators, including parentheses.
57 bool Add(const std::vector<std::string> &lhs, const std::string &op, const std::vector<std::string> &rhs);
58
59 // Check if the given condition values satisfy this set of expressions. First applies
60 // all assignment expressions to create any temporary conditions, then evaluates.
61 bool Test(const ConditionsStore &conditions) const;
62 // Modify the given set of conditions with this ConditionSet.
63 // (Order of operations is like the order of specification: all sibling
64 // expressions are applied, then any and/or nodes are applied.)
65 void Apply(ConditionsStore &conditions) const;
66
67 // Get the names of the conditions that are modified by this ConditionSet.
68 std::set<std::string> RelevantConditions() const;
69
70
71private:
72 // Compare this set's expressions and the union of created and supplied conditions.
73 bool TestSet(const ConditionsStore &conditions, const ConditionsStore &created) const;
74 // Evaluate this set's assignment expressions and store the result in "created" (for use by TestSet).
75 void TestApply(const ConditionsStore &conditions, ConditionsStore &created) const;

105 // A SubExpression results from applying operator-precedence parsing to one side of
106 // an Expression. The operators and tokens needed to recreate the given side are
107 // stored, and can be interleaved to restore the original string. Based on them, a

109 class SubExpression {

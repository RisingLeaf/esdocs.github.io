25// Represents an angle, in degrees. Angles are in "clock" orientation rather
26// than usual mathematical orientation. That is, 0 degrees is up, and angles
27// increase in a clockwise direction. Angles can be efficiently mapped to unit
28// vectors, which also makes rotating a vector an efficient operation.
29class Angle {
30public:
31 // Return a random angle up to the given amount (between 0 and 360).
32 static Angle Random();
33 static Angle Random(double range);
34
35
36public:
37 // The default constructor creates an angle pointing up (zero degrees).
38 Angle() noexcept = default;
39 // Construct an Angle from the given angle in degrees. Allow this conversion
40 // to be implicit to allow syntax like "angle += 30".
41 Angle(double degrees) noexcept;
42 // Construct an angle pointing in the direction of the given vector.
43 explicit Angle(const Point &point) noexcept;
44
45 // Mathematical operators.
46 Angle operator+(const Angle &other) const;
47 Angle &operator+=(const Angle &other);
48 Angle operator-(const Angle &other) const;
49 Angle &operator-=(const Angle &other);
50 Angle operator-() const;
51
52 bool operator==(const Angle &other) const;
53 bool operator!=(const Angle &other) const;
54
55 // Get a unit vector in the direction of this angle.
56 Point Unit() const;
57 // Convert an Angle object to degrees, in the range -180 to 180.
58 double Degrees() const;
59
60 // Return a point rotated by this angle around (0, 0).
61 Point Rotate(const Point &point) const;
62
63
64private:
65 explicit Angle(int32_t angle);
66
67
68private:
69 // The angle is stored as an integer value between 0 and 2^16 - 1. This is
70 // so that any angle can be mapped to a unit vector (a very common operation)
71 // with just a single array lookup. It also means that "wrapping" angles
72 // to the range of 0 to 360 degrees can be done via a bit mask.
73 int32_t angle = 0;
74};

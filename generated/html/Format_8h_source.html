27// Collection of functions for formatting strings for display.
28class Format {
29public:
30 // Function to retrieve a condition's value. Receives a string that contains
31 // the condition name, and the start & size of the substring with the condition
32 // name.
33 using ConditionGetter = std::function<int64_t(const std::string &source, size_t start, size_t size)>;
34
35
36public:
37 // Convert the given number into abbreviated format with a suffix like
38 // "M" for million, "B" for billion, or "T" for trillion. Any number
39 // above 1 quadrillion is instead shown in scientific notation.
40 static std::string Credits(int64_t value);
41 // Convert the given number into abbreviated format as described in Format::Credits,
42 // then attach the ' credit' or ' credits' suffix to it.
43 static std::string CreditString(int64_t value);
44 // Writes the given number into a string,
45 // then attach the ' ton' or ' tons' suffix to it.
46 static std::string MassString(double amount);
47 // Creates a string similar to '<amount> tons of <cargo>'.
48 static std::string CargoString(double amount, const std::string &cargo);
49 // Convert a time in seconds to years/days/hours/minutes/seconds
50 static std::string PlayTime(double timeVal);
51 // Convert the given number to a string, with at most one decimal place.
52 // This is primarily for displaying ship and outfit attributes.
53 static std::string Number(double value);
54 // Format the given value as a number with exactly the given number of
55 // decimal places (even if they are all 0).
56 static std::string Decimal(double value, int places);
57 // Convert numbers to word forms. Capitalize the first letter if at the start of a sentence.
58 static std::string WordForm(int64_t value, bool startOfSentence = false);
59 // Conditionally convert numbers to word forms, based on the Chicago Manual of Style.
60 static std::string ChicagoForm(int64_t value, bool startOfSentence = false);
61 // Conditionally convert numbers to word forms, based on the MLA Style guide.
62 static std::string MLAForm(int64_t value, bool startOfSentence = false);
63 // Convert a string into a number. As with the output of Number(), the
64 // string can have suffixes like "M", "B", etc.
65 static double Parse(const std::string &str);
66 // Replace a set of "keys," which must be strings in the form "<name>", with
67 // a new set of strings, and return the result.
68 static std::string Replace(const std::string &source, const std::map<std::string, std::string> &keys);
69 // Replace all occurrences of "target" with "replacement" in-place.
70 static void ReplaceAll(std::string &text, const std::string &target, const std::string &replacement);
71
72 // Convert a string to title caps or to lower case.
73 static std::string Capitalize(const std::string &str);
74 static std::string LowerCase(const std::string &str);
75
76 // Split a single string into substrings with the given separator.
77 static std::vector<std::string> Split(const std::string &str, const std::string &separator);
78
79 // Finds &[condition] and &[format@condition] in strings and expands them
80 static std::string ExpandConditions(const std::string &source, const ConditionGetter &getter);
81
82 // Function for the "find" dialogs:
83 static int Search(const std::string &str, const std::string &sub);
84};

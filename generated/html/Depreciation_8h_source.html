32// Class for tracking depreciation records, by storing the day on which a given
33// outfit or ship was purchased. Any ship or outfit for which no record exists,
34// for example because it is plunder, counts as full depreciated.
35class Depreciation {
36public:
37 // What fraction of its cost a fully depreciated item has left:
38 static double Full();
39
40
41public:
42 // Load or save depreciation records. Multiple records may be saved in the
43 // player info, under different names (e.g. fleet and stock depreciation).
44 void Load(const DataNode &node);
45 void Save(DataWriter &out, int day) const;
46 // Check if any records have been loaded.
47 bool IsLoaded() const;
48 // If no records have been loaded, initialize with an entire fleet.
49 void Init(const std::vector<std::shared_ptr<Ship>> &fleet, int day);
50
51 // Add a ship, and all its outfits, to the depreciation record.
52 void Buy(const Ship &ship, int day, Depreciation *source = nullptr, bool chassisOnly = false);
53 // Add a single outfit to the depreciation record.
54 void Buy(const Outfit *outfit, int day, Depreciation *source = nullptr);
55
56 // Get the value of an entire fleet.
57 int64_t Value(const std::vector<std::shared_ptr<Ship>> &fleet, int day, bool chassisOnly = false) const;
58 // Get the value of a ship, along with all its outfits.
59 int64_t Value(const Ship &ship, int day) const;
60 // Get the value just of the chassis of a ship.
61 int64_t Value(const Ship *ship, int day, int count = 1) const;
62 // Get the value of an outfit.
63 int64_t Value(const Outfit *outfit, int day, int count = 1) const;
64
65
66private:
67 // "Sell" an item, removing it from the given record and returning the base
68 // day for its depreciation.
69 int Sell(std::map<int, int> &record) const;
70 // Calculate depreciation:
71 double Depreciate(const std::map<int, int> &record, int day, int count = 1) const;
72 double Depreciate(int age) const;
73 // Depreciation of an item for which no record exists. If buying, items
74 // default to no depreciation. When selling, they default to full.
75 double DefaultDepreciation() const;
76
77
78private:
79 // This depreciation record is either a planet's stock or a player's fleet.
80 // If it's the stock, it sells you the most depreciated item first, and once
81 // it runs out of depreciated items all the rest have full price. If it is
82 // your fleet, you sell the least depreciated items first.
83 bool isStock = true;
84 // Check if any data has been loaded.
85 bool isLoaded = false;
86
87 std::map<const Ship *, std::map<int, int>> ships;
88 std::map<const Outfit *, std::map<int, int>> outfits;
89};

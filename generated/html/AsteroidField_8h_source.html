40// Class representing a field of asteroids. The field actually "repeats" every
41// 4096 pixels. That is, an asteroid present at (100, 200) is also present at
42// (4196, 200), (100, 4296), etc. Other games often just "wrap" the asteroids
43// to the screen, meaning that there are no actual asteroids beyond what the
44// player can see, but that means that missiles are not in danger of hitting an
45// asteroid unless they are on screen, and also causes trouble if the screen is
46// resized on the fly. Asteroids never change direction or speed, even if they
47// are hit by a projectile.
48class AsteroidField {
49public:
50 // Constructor, to set up the collision set parameters.
51 AsteroidField();
52
53 // Reset the asteroid field (typically because you entered a new system).
54 void Clear();
55 void Add(const std::string &name, int count, double energy = 1.);
56 void Add(const Minable *minable, int count, double energy, const WeightedList<double> &belts);
57
58 // Move all the asteroids forward one time step, and populate the asteroid and minable collision sets.
59 void Step(std::vector<Visual> &visuals, std::list<std::shared_ptr<Flotsam>> &flotsam, int step);
60 // Draw the asteroid field, with the field of view centered on the given point.
61 void Draw(DrawList &draw, const Point &center, double zoom) const;
62
63 // Check if the given projectile collides with any asteroids. This excludes minables.
64 const std::vector<Collision> &CollideAsteroids(const Projectile &projectile) const;
65 // Check if the given projectile collides with any minables.
66 const std::vector<Collision> &CollideMinables(const Projectile &projectile) const;
67
68 // Get the list of minable asteroids.
69 const std::list<std::shared_ptr<Minable>> &Minables() const;
70
71
72private:
73 // This class represents an asteroid that cannot be destroyed or even
74 // deflected from its trajectory, and that repeats every 4096 pixels.
75 class Asteroid : public Body {
76 public:
77 Asteroid(const Sprite *sprite, double energy);
78
79 void Step();
80 void Draw(DrawList &draw, const Point &center, double zoom) const;
81
82 private:
83 Angle spin;
84 Point size;
85 };

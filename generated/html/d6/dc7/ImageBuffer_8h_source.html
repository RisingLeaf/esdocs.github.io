24// This class stores the raw pixel data from an image, and handles reading that
25// image from disk (so that multiple images can be read and decoded at the same
26// time in different threads). It also handles converting images to
27// premultiplied alpha or additive or half-additive color mixing mode depending
28// on the file name, so that content creators do not have to save the images in
29// some sort of special format.
30class ImageBuffer {
31public:
32 // When initializing a buffer, we know the number of frames but not the size
33 // of them. So, it must be Allocate()d later.
34 ImageBuffer(int frames = 1);
35 ImageBuffer(const ImageBuffer &) = delete;
36 ~ImageBuffer();
37
38 ImageBuffer &operator=(const ImageBuffer &) = delete;
39
40 // Set the number of frames. This must be called before allocating.
41 void Clear(int frames = 1);
42 // Allocate the internal buffer. This must only be called once for each
43 // image buffer; subsequent calls will be ignored.
44 void Allocate(int width, int height);
45
46 int Width() const;
47 int Height() const;
48 int Frames() const;
49
50 const uint32_t *Pixels() const;
51 uint32_t *Pixels();
52
53 const uint32_t *Begin(int y, int frame = 0) const;
54 uint32_t *Begin(int y, int frame = 0);
55
56 void ShrinkToHalfSize();
57
58 // Read a single frame. Return false if an error is encountered - either the
59 // image is the wrong size, or it is not a supported image format.
60 bool Read(const std::string &path, int frame = 0);

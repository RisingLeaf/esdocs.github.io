24// Compare Ts according to the order specified at construction time.
25// Unknown Ts are considered larger than any known one.
26template<class T>
27class ByGivenOrder {
28public:
29 explicit ByGivenOrder(const std::vector<T>& order_)
30 : order(order_)
31 {}
32
33 bool operator()(const T &a, const T &b) const
34 {
35 const auto find_a = std::find(order.begin(), order.end(), a);
36 const auto find_b = std::find(order.begin(), order.end(), b);
37
38 if(find_a == order.end() && find_b == order.end())
39 {
40 // Neither a nor b is a known value. Fall back to default comparison.
41 return (a < b);
42 }
43 else
44 {
45 // Whichever is first in the array is considered smaller.
46 return (find_a < find_b);
47 }
48 }
49
50
51private:
52 const std::vector<T>& order;
53};

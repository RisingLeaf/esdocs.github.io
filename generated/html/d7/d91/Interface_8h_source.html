35// Class representing a user interface, specified in a data file and filled with
36// the contents of an Information object.
37class Interface {
38public:
39 // A destructor is needed to clean up the polymorphic list of elements.
40 Interface() = default;
41 ~Interface();
42
43 void Load(const DataNode &node);
44
45 // Draw this interface. If the given panel is not null, also register any
46 // buttons in this interface with the panel's list of clickable zones.
47 void Draw(const Information &info, Panel *panel = nullptr) const;
48
49 // Get the location of a named point or box.
50 bool HasPoint(const std::string &name) const;

52 Rectangle GetBox(const std::string &name) const;

54 // Get a named value.

75 class Element {
76 public:
77 // State enumeration:
78 static const int INACTIVE = 0;
79 static const int ACTIVE = 1;
80 static const int HOVER = 2;
81
82 public:
83 // Make sure the destructor is virtual, because classes derived from
84 // this one will be used in a polymorphic list.
85 Element() = default;
86 virtual ~Element() = default;
87
88 // Create a new element. The alignment of the interface that contains
89 // this element is used to calculate the element's position.
90 void Load(const DataNode &node, const Point &globalAnchor);
91
92 // Draw this element, relative to the given anchor point. If this is a
93 // button, it will add a clickable zone to the given panel.
94 void Draw(const Information &info, Panel *panel) const;
95
96 // Set the conditions that control when this element is visible and active.
97 // An empty string means it is always visible or active.
98 void SetConditions(const std::string &visible, const std::string &active);
99
100 // Get the bounding rectangle, given the current screen dimensions.
101 Rectangle Bounds() const;
102 // Get the bounding rectangle, treating the Region within the Information as the screen area.
103 Rectangle Bounds(const Information &info) const;
104
105 protected:
106 // Parse the given data line: one that is not recognized by Element
107 // itself. This returns false if it does not recognize the line, either.
108 virtual bool ParseLine(const DataNode &node);
109 // Report the actual dimensions of the object that will be drawn.
110 virtual Point NativeDimensions(const Information &info, int state) const;
111 // Draw this element in the given rectangle.
112 virtual void Draw(const Rectangle &rect, const Information &info, int state) const;
113 // Add any click handlers needed for this element. This will only be
114 // called if the element is visible and active.
115 virtual void Place(const Rectangle &bounds, Panel *panel) const;
116
117 protected:
118 AnchoredPoint from;
119 AnchoredPoint to;
120 Point alignment;
121 Point padding;
122 std::string visibleIf;
123 std::string activeIf;
124 };

139

141 const Sprite *GetSprite(const Information &info, int state) const;
142
143 private:
144 // If a name is given, look up the sprite with that name and draw it.
145 std::string name;
146 // Otherwise, draw a sprite. Which sprite is drawn depends on the current
147 // state of this element: inactive, active, or hover.
148 const Sprite *sprite[3] = {nullptr, nullptr, nullptr};
149 // If this flag is set, draw the sprite as an outline:
150 bool isOutline = false;
151 // Store whether the outline should be colored.
152 bool isColored = false;

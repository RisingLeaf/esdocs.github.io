29// Class representing messages that should be shown to the user. The messages
30// gradually fade as the game steps forward, so each one must remember the game
31// step when it came into being. If a new message is added that exactly matches
32// an old one, the old version is removed before the new one is added; this is
33// to keep repeated messages from filling up the whole screen.
34class Messages {
35public:
36 enum class Importance : uint_least8_t {
37 Highest,
38 High,
39 Info,
40 Daily,
41 Low
42 };
43
44 class Entry {
45 public:
46 Entry() = default;
47 Entry(int step, const std::string &message, Importance importance)
48 : step(step), message(message), importance(importance) {}
49
50 int step;
51 std::string message;
52 Importance importance;
53 };
54
55public:
56 // Add a message to the list along with its level of importance
57 static void Add(const std::string &message, Importance importance = Importance::Low);
58 // Add a message to the log. For messages meant to be shown
59 // also on the main panel, use Add instead.
60 static void AddLog(const std::string &message, Importance importance = Importance::Low);
61
62 // Get the messages for the given game step. Any messages that are too old
63 // will be culled out, and new ones that have just been added will have
64 // their "step" set to the given value.
65 static const std::vector<Entry> &Get(int step);
66 static const std::deque<std::pair<std::string, Messages::Importance>> &GetLog();
67
68 // Reset the messages (i.e. because a new game was loaded).
69 static void Reset();
70
71 // Get color that should be used for drawing messages of given importance.
72 static const Color *GetColor(Importance importance, bool isLogPanel);
73};

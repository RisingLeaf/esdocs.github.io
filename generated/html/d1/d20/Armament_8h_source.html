35// This class handles the logic for a ship's set of weapons. All weapons of the
36// same type coordinate their fire with each other, either firing in clusters
37// (if the projectiles are vulnerable to anti-missile) or in a "stream" where
38// the guns take turns firing. Instead of firing straight, guns (that is, non-
39// turreted weapons) fire aimed slightly inward in a convergence pattern so
40// that even if the guns are spaced out horizontally on the ship, their
41// projectiles will nearly meet at the end of their range. This class also
42// handles turrets, which aim automatically and take into account the target's
43// distance away and velocity relative to the ship that is firing.
44class Armament {
45public:
46 // Add a gun or turret hard-point.
47 void AddGunPort(const Point &point, const Angle &angle, bool isParallel, bool isUnder, const Outfit *outfit = nullptr);
48 void AddTurret(const Point &point, bool isUnder, const Outfit *outfit = nullptr);
49 // This must be called after all the outfit data is loaded. If you add more
50 // of a given weapon than there are slots for it, the extras will not fire.
51 // But, the "gun ports" attribute should keep that from happening. To
52 // remove a weapon, just pass a negative value here.
53 int Add(const Outfit *outfit, int count = 1);
54 // Call this once all the outfits have been loaded to make sure they are all
55 // set up properly (even the ones that were pre-assigned to a hardpoint).
56 void FinishLoading();
57 // Reload all weapons (because a day passed in-game).
58 void ReloadAll();
59 // Uninstall all weapons (because the weapon outfits have potentially changed).
60 void UninstallAll();
61
62 // Swap the weapons in the given two hardpoints.
63 void Swap(unsigned first, unsigned second);
64
65 // Access the array of weapon hardpoints.
66 const std::vector<Hardpoint> &Get() const;
67 int GunCount() const;
68 int TurretCount() const;
69 // Determine the ammunition used by this armament that can be resupplied (i.e. is not self-uninstalling).
70 std::set<const Outfit *> RestockableAmmo() const;
71
72 // Adjust the aim of the turrets.
73 void Aim(const FireCommand &command);
74 // Fire the given weapon, if it is ready. If it did not fire because it is
75 // not ready, return false.
76 void Fire(unsigned index, Ship &ship, std::vector<Projectile> &projectiles, std::vector<Visual> &visuals, bool jammed);
77 // Fire the given anti-missile system.
78 bool FireAntiMissile(unsigned index, Ship &ship, const Projectile &projectile,
79 std::vector<Visual> &visuals, bool jammed);
80 // Fire the given tractor beam.
81 bool FireTractorBeam(unsigned index, Ship &ship, const Flotsam &flotsam,
82 std::vector<Visual> &visuals, bool jammed);
83
84 // Update the reload counters.
85 void Step(const Ship &ship);
86
87
88private:
89 // Check if the given hardpoint index is valid. Jam the hardpoint if it needs jammed.
90 // Returns false if the index is invalid or the hardpoint jammed.
91 bool CheckHardpoint(unsigned index, bool jammed = false);
92
93
94private:
95 // Note: the Armament must be copied when an instance of a Ship is made, so
96 // it should not hold any pointers specific to one ship (including to
97 // elements of this Armament itself).
98 std::map<const Outfit *, int> streamReload;
99 std::vector<Hardpoint> hardpoints;
100};

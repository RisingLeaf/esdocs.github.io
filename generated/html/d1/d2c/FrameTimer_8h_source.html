23// Class to be used for enforcing a certain frame rate. This ensures that the
24// frames are not drawn faster than a certain speed, but if the calculations or
25// the graphics cannot keep up it will allow things to go slower for a few frames
26// without trying to "catch up" by making the subsequent frame faster.
27class FrameTimer {
28public:
29 // Create a timer that is just responsible for measuring the time that
30 // elapses until Time() is called.
31 FrameTimer();
32 // Create a frame timer that will space frames out at exactly the given FPS,
33 // _unless_ a frame takes too long by at least the given lag, in which case
34 // the next frame happens immediately but no "catch-up" is done.
35 explicit FrameTimer(int fps, int maxLagMsec = 5);
36
37 // Wait until the next frame should begin.
38 void Wait();
39 // Find out how long it has been since this timer was created, in seconds.
40 double Time() const;
41
42 // Change the frame rate (for viewing in slow motion).
43 void SetFrameRate(int fps);
44
45
46private:
47 // Calculate when the next frame should begin.
48 void Step();
49
50
51private:
52 std::chrono::steady_clock::time_point next;
53 std::chrono::steady_clock::duration step;
54 std::chrono::steady_clock::duration maxLag;
55};

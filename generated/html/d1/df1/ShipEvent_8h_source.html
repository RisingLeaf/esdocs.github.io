26// A ShipEvent is anything of significance that one ship does to another. These
27// events might change one government's attitude toward the player, or may
28// result in a certain message or dialog being displayed.
29class ShipEvent {
30public:
31 enum {
32 // This is a "null" event.
33 NONE = 0,
34 // This ship did something good for the given ship.
35 ASSIST = (1 << 0),
36 // This ship scanned the given ship's cargo. This is not necessarily an
37 // act of aggression, but it implies mistrust or underhanded intentions.
38 // Also, a mission may fail if a ship of a certain government scans your
39 // cargo and discovers you are carrying contraband.
40 SCAN_CARGO = (1 << 1),
41 // This ship scanned the given ship's outfits. (If it turns out the
42 // outfits include something illegal, this may result in a fine or an
43 // outright attack on the ship that was scanned.)
44 SCAN_OUTFITS = (1 << 2),
45 // This ship damaged the given ship while not currently being an enemy
46 // of that ship's government; this will result in temporary animosities
47 // between the two governments. If a ship is "forbearing," it can only
48 // be "provoked" if its shields are below 90%.
49 // Some governments are provoked by starting a scan.
50 PROVOKE = (1 << 3),
51 // This ship disabled the given ship. This will have a permanent effect
52 // on your reputation with the given government. This event is generated
53 // when a ship takes damage that switches it to being disabled.
54 DISABLE = (1 << 4),
55 // This ship boarded the given ship. This may either be an attempt to
56 // render assistance, or an attempt to capture the ship.
57 BOARD = (1 << 5),
58 // This ship captured the given ship.
59 CAPTURE = (1 << 6),
60 // This ship destroyed the given ship. If your projectiles hit a ship
61 // that is already exploding, that does not generate a "destroy" event;
62 // this is only for the one projectile that caused the explosion.
63 DESTROY = (1 << 7),
64 // This is a crime that is so bad that it not only has a negative effect
65 // on your reputation, but entirely wipes out any positive reputation
66 // you had with the given government, first.
67 ATROCITY = (1 << 8),
68 // This ship just jumped into a different system.
69 JUMP = (1 << 9),
70 // This ship just met another ship for the first time
71 ENCOUNTER = (1 << 10),
72 };

75public:
76 ShipEvent(const Government *actor, const std::shared_ptr<Ship> &target, int type);
77 ShipEvent(const std::shared_ptr<Ship> &actor, const std::shared_ptr<Ship> &target, int type);

90 const Government *targetGovernment = nullptr;

92};

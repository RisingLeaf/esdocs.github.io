27// Class for queuing up tasks that are to be executed in parallel by using
28// every thread available on the executing CPU.
29// The queue is also responsible to execute follow-up tasks that need to
30// executed after the async task, for example uploading a loaded to the GPU
31// (which needs to happen on the main thread on OpenGL).
32class TaskQueue {
33public:
34 // An internal structure representing a task to execute.
35 struct Task {
36 TaskQueue *queue;
37
38 // The function to execute in parallel.
39 std::function<void()> async;
40 // If specified, this function is called in the main thread after
41 // the function above has finished executing.
42 std::function<void()> sync;
43
44 // The pointer to this task's future stored in the queue.
45 std::list<std::shared_future<void>>::const_iterator futureIt;
46
47 std::promise<void> futurePromise;
48 };
49
50 // The maximum amount of sync tasks to execute in one go.
51 static constexpr int MAX_SYNC_TASKS = 100;
52
53public:
54 // Initialize the threads used to execute the tasks.
55 TaskQueue() = default;
56 TaskQueue(const TaskQueue &) = delete;
57 TaskQueue &operator=(const TaskQueue &) = delete;
58 ~TaskQueue();
59
60 // Queue a function to execute in parallel, with an another optional function that
61 // will get executed on the main thread after the first function finishes.
62 // Returns a future representing the future result of the async call. Ignores
63 // any main thread task that still need to be executed!
64 std::shared_future<void> Run(std::function<void()> asyncTask, std::function<void()> syncTask = {});
65
66 // Process any tasks to be scheduled to be executed on the main thread.
67 void ProcessSyncTasks();
68
69 // Waits for all of this queue's task to finish. Ignores any sync tasks to be processed.
70 void Wait();

73private:

89};

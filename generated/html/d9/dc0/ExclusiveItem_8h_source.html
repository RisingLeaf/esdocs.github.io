24// This class provides "data template" classes with abstracted access to an object that
25// is either a reference to a shared, "stock" data or is a locally customized instance.
26template <class Type>
27class ExclusiveItem {
28public:
29 ExclusiveItem() = default;
30
31 explicit ExclusiveItem(const Type *item) : stockItem(item) {}
32 explicit ExclusiveItem(Type &&item) : item(std::move(item)) {}
33
34 ExclusiveItem(ExclusiveItem&&) = default;
35 ExclusiveItem &operator=(ExclusiveItem&&) = default;
36 ExclusiveItem(const ExclusiveItem&) = default;
37 ExclusiveItem &operator=(const ExclusiveItem&) = default;
38
39 bool IsStock() const noexcept { return stockItem; }
40
41 const Type *operator->() const noexcept { return stockItem ? stockItem : std::addressof(item); }
42 const Type &operator*() const noexcept { return stockItem ? *stockItem : item; }
43
44 bool operator==(const ExclusiveItem &other) const { return this->operator*() == other.operator*(); }
45 bool operator!=(const ExclusiveItem &other) const { return !(this->operator*() == other.operator*()); }

48private:
49 const Type *stockItem = nullptr;
50 Type item;
51};

31// A CollisionSet allows efficient collision detection by splitting space up
32// into a grid and keeping track of which objects are in each grid cell. A check
33// for collisions can then only examine objects in certain cells.
34class CollisionSet {
35public:
36 // Initialize a collision set. The cell size and cell count should both be
37 // powers of two; otherwise, they are rounded down to a power of two.
38 CollisionSet(unsigned cellSize, unsigned cellCount, CollisionType collisionType);
39
40 // Clear all objects in the set. Specify which engine step we are on, so we
41 // know what animation frame each object is on.
42 void Clear(int step);
43 // Add an object to the set.
44 void Add(Body &body);
45 // Finish adding objects (and organize them into the final lookup table).
46 void Finish();
47
48 // Get all possible collisions for the given projectile. Collisions are not necessarily
49 // sorted by distance.
50 const std::vector<Collision> &Line(const Projectile &projectile) const;
51
52 // Get all possible collisions along a line. Collisions are not necessarily sorted by
53 // distance.
54 const std::vector<Collision> &Line(const Point &from, const Point &to,
55 const Government *pGov = nullptr, const Body *target = nullptr) const;
56
57 // Get all objects within the given range of the given point.
58 const std::vector<Body *> &Circle(const Point &center, double radius) const;
59 // Get all objects touching a ring with a given inner and outer range
60 // centered at the given point.
61 const std::vector<Body *> &Ring(const Point &center, double inner, double outer) const;
62
63 // Get all objects within this collision set.
64 const std::vector<Body *> &All() const;
65
66
67private:
68 class Entry {
69 public:
70 Entry() = default;
71 Entry(Body *body, unsigned seenIndex, int x, int y) : body(body), seenIndex(seenIndex), x(x), y(y) {}
72
73 Body *body;
74 unsigned seenIndex;
75 int x;
76 int y;
77 };

79
80private:
81 // The type of collisions this CollisionSet is responsible for.
82 CollisionType collisionType;
83
84 // The size of individual cells of the grid.
85 unsigned CELL_SIZE;
86 unsigned SHIFT;
87 unsigned CELL_MASK;
88
89 // The number of grid cells.
90 unsigned CELLS;
91 unsigned WRAP_MASK;
92
93 // The current game engine step.
94 int step;
95
96 // Vectors to store the objects in the collision set.
97 std::vector<Body *> all;
98 std::vector<Entry> added;
99 std::vector<Entry> sorted;
100 // After Finish(), counts[index] is where a certain bin begins.
101 std::vector<unsigned> counts;
102
103 // Vector for returning the result of a circle query.
104 mutable std::vector<Body *> circleResult;
105 // Vector for returning the result of a line query.
106 mutable std::vector<Collision> lineResult;
107
108 // Keep track of which objects we've already considered
109 mutable std::vector<unsigned> seen;
110 mutable unsigned seenEpoch = 0;
111};

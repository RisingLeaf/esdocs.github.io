21// Smoothly change a variable from one value to another. Used to smooth out
22// scrolling and panning.
23template <typename T>
24class Animate
25{
26public:
27 virtual ~Animate() = default;
28 // Set the next target value of this variable, linearly interpolated along
29 // the given number of frames.
30 virtual void Set(const T &current, int steps = 5);
31
32 // Reset the pending number of frames to zero. This makes the interpolated
33 // value jump straight to the target value.
34 void EndAnimation();
35 // Compute the next interpolated value. This needs to be called once per frame.
36 void Step();
37
38 // Returns the interpolated value.
39 const T &AnimatedValue() const;
40 // Returns the actual value.
41 const T &Value() const;
42 // Synonym for Value().
43 operator const T &() const;
44
45 // Shortcut mathematical operators for convenience.
46 Animate &operator=(const T &v);
47 Animate &operator+=(const T &v);
48 Animate &operator-=(const T &v);
49
50
51private:
52 int steps = 0;
53 T current{};
54 T target{};
55
56};

77template <typename T>
78void Animate<T>::Step()
79{
80 if(steps <= 0)
81 current = target;
82 else
83 {
84 T delta = (target - current) / steps;
85 current += delta;
86 --steps;
87 }
88}

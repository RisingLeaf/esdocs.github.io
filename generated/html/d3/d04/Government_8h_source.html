43// Class representing a government. Each ship belongs to some government, and
44// attacking that ship will provoke its ally governments and reduce your
45// reputation with them, but increase your reputation with that ship's enemies.
46// The ships for each government are identified by drawing them with a different
47// color "swizzle." Some government's ships can also be easier or harder to
48// bribe than others.
49class Government {
50public:
51 // Default constructor.
52 Government();
53
54 // Load a government's definition from a file.
55 void Load(const DataNode &node);
56
57 // Get the display name of this government.
58 const std::string &GetName() const;
59 // Set / Get the name used for this government in the data files.
60 void SetName(const std::string &trueName);
61 const std::string &GetTrueName() const;
62 // Get the color swizzle to use for ships of this government.
63 int GetSwizzle() const;
64 // Get the color to use for displaying this government on the map.
65 const Color &GetColor() const;
66
67 // Get the government's initial disposition toward other governments or
68 // toward the player.
69 double AttitudeToward(const Government *other) const;
70 double InitialPlayerReputation() const;
71 // Get the amount that your reputation changes for the given offense against the given government.
72 // The given value should be a combination of one or more ShipEvent values.
73 // Returns 0 if the Government is null.
74 double PenaltyFor(int eventType, const Government *other) const;
75 // In order to successfully bribe this government you must pay them this
76 // fraction of your fleet's value. (Zero means they cannot be bribed.)
77 double GetBribeFraction() const;
78 // This government will fine you the given fraction of the maximum fine for
79 // carrying illegal cargo or outfits. Zero means they will not fine you.
80 double GetFineFraction() const;
81 bool Trusts(const Government *other) const;
82 // A government might not exercise the ability to perform scans or fine
83 // the player in every system.
84 bool CanEnforce(const System *system) const;
85 bool CanEnforce(const Planet *planet) const;
86 // Get the conversation that will be shown if this government gives a death
87 // sentence to the player (for carrying highly illegal cargo).
88 const Conversation *DeathSentence() const;
89
90 // Get a hail message (which depends on whether this is an enemy government
91 // and if the ship is disabled).
92 std::string GetHail(bool isDisabled) const;
93 // Find out if this government speaks a different language.
94 const std::string &Language() const;
95 // Find out if this government should send custom hails even if the player does not know its language.
96 bool SendUntranslatedHails() const;
97 // Pirate raids in this government's systems use these fleet definitions. If
98 // it is empty, there are no pirate raids.
99 // The second attribute denotes the minimal and maximal attraction required for the fleet to appear.
100 const std::vector<RaidFleet> &RaidFleets() const;
101
102 // Check if, according to the politics stored by GameData, this government is
103 // an enemy of the given government right now.
104 bool IsEnemy(const Government *other) const;
105 // Check if this government is an enemy of the player.
106 bool IsEnemy() const;
107
108 // Below are shortcut functions which actually alter the game state in the
109 // Politics object, but are provided as member functions here for clearer syntax.
110
111 // Check if this is the player government.
112 bool IsPlayer() const;
113 // Commit the given "offense" against this government (which may not
114 // actually consider it to be an offense). This may result in temporary
115 // hostilities (if the even type is PROVOKE), or a permanent change to your
116 // reputation.
117 void Offend(int eventType, int count = 1) const;
118 // Bribe this government to be friendly to you for one day.
119 void Bribe() const;
120 // Check to see if the player has done anything they should be fined for.
121 // Each government can only fine you once per day.
122 std::string Fine(PlayerInfo &player, int scan = 0, const Ship *target = nullptr, double security = 1.) const;
123 // Check to see if the items are condemnable (atrocities) or warrant a fine.
124 bool Condemns(const Outfit *outfit) const;
125 bool Condemns(const Ship *ship) const;
126 // Returns the fine for given item for this government.
127 int Fines(const Outfit *outfit) const;
128 int Fines(const Ship *ship) const;
129 // Check if given ship has illegal outfits or cargo.
130 bool FinesContents(const Ship *ship) const;
131
132 // Get or set the player's reputation with this government.
133 double Reputation() const;
134 double ReputationMax() const;
135 double ReputationMin() const;
136 void AddReputation(double value) const;
137 void SetReputation(double value) const;

139 // Get the government's crew attack/defense values
140 double CrewAttack() const;
141 double CrewDefense() const;
142
143 bool IsProvokedOnScan() const;
144
145 // Determine if ships from this government can travel to the given system or planet.

147 bool IsRestrictedFrom(const Planet &planet) const;

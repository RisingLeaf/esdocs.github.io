30// Class that contains storage for conditions. Those conditions can be
31// set directly in internal storage of this class (primary conditions)
32// and those conditions can also be provided from other locations in the
33// code (derived conditions).
34//
35// The derived conditions are typically provided "on demand" from outside
36// this storage class. Some of those derived conditions could be read-only
37// and in a number of cases the conditions might be converted from other
38// data types than int64_t (for example double, float or even complex
39// formulae).
40class ConditionsStore {
41public:
42 // Forward declaration, needed to make ConditionEntry a friend of the
43 // DerivedProvider.
44 class ConditionEntry;
45
46 // Class for DerivedProviders, the (lambda) functions that provide access
47 // to the derived conditions are registered in this class.
48 class DerivedProvider {
49 friend ConditionsStore;
50 friend ConditionEntry;
51
52 public:
53 // Functions to set the lambda functions for accessing the conditions.
54 void SetGetFunction(std::function<int64_t(const std::string &)> newGetFun);
55 void SetHasFunction(std::function<bool(const std::string &)> newHasFun);
56 void SetSetFunction(std::function<bool(const std::string &, int64_t)> newSetFun);
57 void SetEraseFunction(std::function<bool(const std::string &)> newEraseFun);
58
59 public:
60 // This is intended as a private constructor, only to be called from within
61 // ConditionsStore. But we need to keep it public because of how the
62 // DerivedProviders are emplaced in the providers-map-variable.
63 DerivedProvider(const std::string &name, bool isPrefixProvider);
64
65 private:
66 std::string name;
67 bool isPrefixProvider;
68
69 // Lambda functions for accessing the derived conditions, with some sensible
70 // default implementations;
71 std::function<int64_t(const std::string &)> getFunction = [](const std::string &name) { return 0; };
72 std::function<bool(const std::string &)> hasFunction = [](const std::string &name) { return true; };
73 std::function<bool(const std::string &, int64_t)> setFunction = [](const std::string &name, int64_t value) {
74 return false; };
75 std::function<bool(const std::string &)> eraseFunction = [](const std::string &name) { return false; };
76 };

79 // Storage entry for a condition. Can act as a int64_t proxy when operator[] is used for access
80 // to conditions in the ConditionsStore.
81 class ConditionEntry {
82 friend ConditionsStore;
83
84 public:
85 // int64_t proxy helper functions. Those functions allow access to the conditions
86 // using `operator[]` on ConditionsStore.
87 operator int64_t() const;
88 ConditionEntry &operator=(int64_t val);
89 ConditionEntry &operator++();
90 ConditionEntry &operator--();
91 ConditionEntry &operator+=(int64_t val);
92 ConditionEntry &operator-=(int64_t val);
93
94 private:
95 int64_t value = 0;
96 DerivedProvider *provider = nullptr;
97 // The full keyname for condition we want to access. This full keyname is required
98 // when accessing prefixed providers, because such providers will only know the prefix
99 // part of the key.
100 std::string fullKey;
101 };

147 const ConditionEntry *GetEntry(const std::string &name) const;
148 bool VerifyProviderLocation(const std::string &name, DerivedProvider *provider) const;
149
150
151
152private:
153 // Storage for both the primary conditions as well as the providers.

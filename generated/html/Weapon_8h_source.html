38// Class representing all the characteristics of a weapon, including sprites and
39// effects, sounds, icons, ammo, submunitions, and other attributes. Storing
40// these parameters in a separate class keeps each Projectile from needing a
41// copy of them, and storing them as class variables instead of in a map of
42// string to double significantly reduces access time.
43class Weapon {
44public:
45 struct Submunition{
46 Submunition() noexcept = default;
47 explicit Submunition(const Weapon *weapon, std::size_t count) noexcept
48 : weapon(weapon), count(count) {};
49
50 const Weapon *weapon = nullptr;
51 std::size_t count = 0;
52 // The angular offset from the source projectile, relative to its current facing.
53 Angle facing;
54 // The base offset from the source projectile's position, relative to its current facing.
55 Point offset;
56 };

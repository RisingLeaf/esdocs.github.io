30// Class representing a UI (i.e. a series of Panels stacked on top of each other,
31// with Panels added or removed in response to user events). This handles events
32// and passing them on to whatever panel is on top, and drawing of the panels
33// starting with whichever one is on the bottom.
34class UI {
35public:
36 // Handle an event. The event is handed to each panel on the stack until one
37 // of them handles it. If none do, this returns false.
38 bool Handle(const SDL_Event &event);
39
40 // Step all the panels forward (advance animations, move objects, etc.).
41 void StepAll();
42 // Draw all the panels.
43 void DrawAll();
44
45 // Add the given panel to the stack. If you do not want a panel to be
46 // deleted when it is popped, save a copy of its shared pointer elsewhere.
47 void Push(Panel *panel);
48 void Push(const std::shared_ptr<Panel> &panel);
49 // Remove the given panel from the stack (if it is in it). The panel will be
50 // deleted at the start of the next time Step() is called, so it is safe for
51 // a panel to Pop() itself.
52 void Pop(const Panel *panel);
53 // Remove the given panel and every panel that is higher in the stack.
54 void PopThrough(const Panel *panel);
55
56 // Check whether the given panel is on top, i.e. is the active one, out of
57 // all panels that are already drawn on this step.
58 bool IsTop(const Panel *panel) const;
59 // Get the top panel, out of all possible panels, including ones not yet drawn.
60 std::shared_ptr<Panel> Top() const;
61
62 // Delete all the panels and clear the "done" flag.
63 void Reset();
64 // Get the lower-most panel.
65 std::shared_ptr<Panel> Root() const;
66
67 // If the player enters the game, enable saving the loaded file.
68 void CanSave(bool canSave);
69 bool CanSave() const;
70 // Tell the UI to quit.
71 void Quit();
72 // Check if it is time to quit.
73 bool IsDone() const;
74 // Check if there are no panels left.
75 bool IsEmpty() const;
76
77 // Get the current mouse position.
78 static Point GetMouse();
79
80
81private:
82 // If a push or pop is queued, apply it.
83 void PushOrPop();
84
85
86private:
87 // Whether the player has taken actions that enable us to save the game.
88 bool canSave = false;
89 // Whether the player has requested the game to shut down.
90 bool isDone = false;
91
92 std::vector<std::shared_ptr<Panel>> stack;
93 std::vector<std::shared_ptr<Panel>> toPush;
94 std::vector<const Panel *> toPop;
95};

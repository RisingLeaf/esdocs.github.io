23// This class allows you to set a scroll value and provides animation
24// interpolation between the old and new values, while clamping the value to a
25// suitable range. The Animate methods will return negative values, as they are
26// meant to be added as an offset to the draw position.
27template <typename T>
28class ScrollVar: public Animate<T>
29{
30public:
31 ScrollVar() = default;
32 ScrollVar(const T &maxVal, const T &displaySize);
33
34 // Set the maximum size of the scroll content.
35 void SetMaxValue(const T &value);
36 // Get the maximum size of the scroll content.
37 const T &MaxValue() const;
38 // Set the size of the displayable scroll area.
39 void SetDisplaySize(const T &size);
40 // Returns true if scroll buttons are needed.
41 bool Scrollable() const;
42 // Returns true if the value is at the minimum.
43 bool IsScrollAtMin() const;
44 // Returns true if the value is at the maximum.
45 bool IsScrollAtMax() const;
46 // Modifies the scroll value by dy, then clamps it to a suitable range.
47 void Scroll(const T &dy, int steps = 5);
48
49 // Sets the scroll value directly, then clamps it to a suitable range.
50 virtual void Set(const T &current, int steps = 5) override;
51
52 // Shortcut mathematical operators for convenience.
53 ScrollVar &operator=(const T &v);
54
55
56private:
57 // Makes sure the animation value stays in range.
58 void Clamp(int steps);
59
60 T maxVal{};
61 T displaySize{};
62};

141template <typename T>
142void ScrollVar<T>::Clamp(int steps)
143{
144 int maxScroll = maxVal - displaySize;
145 if(this->Value() < T{} || maxVal < displaySize)
146 Animate<T>::Set(T{}, steps);
147 else if(this->Value() > maxScroll)
148 Animate<T>::Set(maxScroll, steps);
149}

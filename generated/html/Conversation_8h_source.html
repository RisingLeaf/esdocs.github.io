34// Class representing a conversation, generally occurring when the player is asked to
35// accept or decline a mission. The conversation can take different paths depending
36// on what responses you choose, leading you to accept, decline, or (rarely) to be
37// killed. A conversation can also branch based on various condition flags that
38// are set for the player, or even trigger various changes to the game's state.
39class Conversation {
40public:
41 // The possible outcomes of a conversation:
42 static const int ACCEPT = -1;
43 static const int DECLINE = -2;
44 static const int DEFER = -3;
45 // These 3 options force the player to TakeOff (if landed), or cause
46 // the boarded NPCs to explode, in addition to respectively duplicating
47 // the above mission outcomes.
48 static const int LAUNCH = -4;
49 static const int FLEE = -5;
50 static const int DEPART = -6;
51 // The player may simply die (if landed on a planet or captured while
52 // in space), or the flagship might also explode.
53 static const int DIE = -7;
54 static const int EXPLODE = -8;
55
56 // Check whether the given conversation outcome is one that forces the
57 // player to immediately depart.
58 static bool RequiresLaunch(int outcome);
59
60public:
61 Conversation() = default;
62 // Construct and Load() at the same time.
63 Conversation(const DataNode &node);
64 // Read or write to files.
65 void Load(const DataNode &node);
66 void Save(DataWriter &out) const;
67 // Check if any data is loaded in this conversation object.
68 bool IsEmpty() const noexcept;
69 // Check if this conversation includes a name prompt.
70 bool IsValidIntro() const noexcept;
71 // Check if the actions in this conversation are valid.
72 std::string Validate() const;
73
74 // Generate a new conversation from this template, filling in any text
75 // substitutions and instantiating any actions.
76 Conversation Instantiate(std::map<std::string, std::string> &subs, int jumps = 0, int payload = 0) const;
77
78 // The beginning of the conversation is node 0. Some nodes have choices for
79 // the user to select; others just automatically continue to another node.
80 // Nodes may also display images or include conditional branches.
81 bool IsChoice(int node) const;
82 // Some choices have conditions in each option. If all options are disabled,
83 // the choice cannot be shown.
84 bool HasAnyChoices(const ConditionsStore &vars, int node) const;
85 // If the given node is a choice node, check how many choices it offers.
86 int Choices(int node) const;
87 // Check if the given conversation node is a conditional branch.
88 bool IsBranch(int node) const;
89 // Check if the given conversation node performs an action.
90 bool IsAction(int node) const;
91 const ConditionSet &Conditions(int node) const;
92 const GameAction &GetAction(int node) const;
93 const std::string &Text(int node, int element = 0) const;
94 const Sprite *Scene(int node) const;
95 // Find out where the conversation goes if the given option is chosen.
96 int NextNodeForChoice(int node, int element = 0) const;
97 // Go to the next node of the conversation, ignoring any choices.
98 int StepToNextNode(int node) const;
99 // Returns whether the given node should be displayed.
100 // Returns false if:
101 // - The node (or element) is out of range.
102 // - The node is not a choice node, and element is non-zero.
103 // - The node (or element) has conditions and those conditions are not met.
104 // and true otherwise.
105 bool ShouldDisplayNode(const ConditionsStore &vars, int node, int element = 0) const;
106 // Returns true if the given node index is in the range of valid nodes for
107 // this Conversation.
108 // Note: This function only considers actual Conversation nodes to be valid
109 // choices. The negative "outcome" values (ACCEPT, DEFER, etc.) are special
110 // sentinel values, rather than indices of Conversation nodes, and are
111 // therefore considered invalid by this function.
112 bool NodeIsValid(int node) const;
113 // Returns true if the given node index is in the range of valid nodes for
114 // this Conversation *and* the given element index is in the range of valid
115 // elements for the given node.
116 bool ElementIsValid(int node, int element) const;
117
118
119private:
120 // This serves multiple purposes:
121 // - In a regular text node, there's exactly one of these. It contains the
122 // text data, the index of the next node to unconditionally visit, and,
123 // optionally, a condition set which, if not met, prevents the text from
124 // being displayed (without affecting which node is processed next).
125 // - In a choice node, there's one of these for each possible choice,
126 // containing the text to display, the node the choice leads to, and,
127 // optionally, the conditions under which to offer the choice.
128 // - In a branch node, there's two of these. The first one contains the
129 // condition for the branch. If the condition is met, the "next" member
130 // of the first element is followed. If it's not met, it's the second
131 // element whose "next" member is followed.
132 class Element {
133 public:
134 explicit Element(std::string text, int next)
135 : text(std::move(text)), next(next) {}
136 // The text to display:
137 std::string text;
138 // The next node to visit:
139 int next;
140 // Conditions for displaying the text:
141 ConditionSet conditions;
142 };
143
144 // The conversation is a network of "nodes" that you travel between by
145 // making choices (or by automatic branches that depend on the condition
146 // variable values for the current player).
147 class Node {
148 public:
149 // Construct a new node. Each paragraph of conversation that involves no
150 // choice can be merged into what came before it, to simplify things.
151 explicit Node(bool isChoice = false) noexcept : isChoice(isChoice), canMergeOnto(!isChoice) {}
152
153 // The condition expressions that determine the next node to load, or
154 // whether to display.
155 ConditionSet conditions;
156 // Tasks performed when this node is reached.
157 GameAction actions;
158 // See Element's comment above for what this actually entails.
159 std::vector<Element> elements;
160 // This distinguishes "choice" nodes from "branch" or text nodes. If
161 // this value is false, a one-element node is considered text, and a
162 // node with more than one element is considered a "branch".
163 bool isChoice;
164 // Keep track of whether it's possible to merge future nodes onto this.
165 bool canMergeOnto;
166
167 // Image that should be shown along with this text.
168 const Sprite *scene = nullptr;
169 };
170
171
172private:
173 // Parse the children of the given node to see if they contain any "gotos"
174 // or "conditions." If so, link them up properly. Return true if gotos or
175 // conditions were found.
176 bool LoadDestinations(const DataNode &node);
177 // Parse the children to see if there is a condition.
178 bool HasDisplayRestriction(const DataNode &node);
179 // Add a label, pointing to whatever node is created next.
180 void AddLabel(const std::string &label, const DataNode &node);
181 // Set up a "goto". Depending on whether the named label has been seen yet
182 // or not, it is either resolved immediately or added to the unresolved set.
183 void Goto(const std::string &label, int node, int element = 0);
184 // Add an "empty" node. It will contain one empty line of text, with its
185 // goto link set to fall through to the next node.
186 void AddNode();
187
188
189private:
190 // While parsing the conversation, keep track of what labels link to what
191 // nodes. If a name appears in a goto before that label appears, remember
192 // what node and what element it appeared at in order to link it up later.
193 std::map<std::string, int> labels;
194 std::multimap<std::string, std::pair<int, int>> unresolved;
195 // The actual conversation data:
196 std::vector<Node> nodes;
197};

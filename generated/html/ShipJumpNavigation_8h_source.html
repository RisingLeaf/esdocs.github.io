29// A class representing the jump capabilities of a ship. Calculates and caches a ship's
30// jump methods, costs, and distances.
31class ShipJumpNavigation {
32public:
33 static const double DEFAULT_HYPERDRIVE_COST;
34 static const double DEFAULT_SCRAM_DRIVE_COST;
35 static const double DEFAULT_JUMP_DRIVE_COST;

37
38public:
39 ShipJumpNavigation() = default;

41 // Calibrate this ship's jump navigation information, caching its jump costs, range, and capabilities.
42 void Calibrate(const Ship &ship);
43 // Recalibrate jump costs for this ship, but only if necessary.
44 void Recalibrate(const Ship &ship);
45
46 // Pass the current system that the ship is in to the navigation.
47 void SetSystem(const System *system);
48
49 // Get the amount of fuel that would be expended to jump to the destination. If the destination is
50 // nullptr then return the maximum amount of fuel that this ship could expend in one jump.
51 double JumpFuel(const System *destination = nullptr) const;
52 // Get the maximum distance that this ship can jump.
53 double JumpRange() const;
54 // Get the cost of making a jump of the given type (if possible). Returns 0 if the jump can't be made.
55 double HyperdriveFuel() const;
56 double JumpDriveFuel(double distance = 0.) const;

58 // If no jump method is possible, returns JumpType::None with a jump cost of 0.
59 std::pair<JumpType, double> GetCheapestJumpType(const System *destination) const;
60 // Get the cheapest jump method between the two given systems.
61 std::pair<JumpType, double> GetCheapestJumpType(const System *from, const System *to) const;
62
63 // Get if this ship can make a hyperspace or jump drive jump directly from one system to the other.
64 bool CanJump(const System *from, const System *to) const;
65
66 // Check what jump methods this ship has.
67 bool HasHyperdrive() const;
68 bool HasScramDrive() const;
69 bool HasJumpDrive() const;

71
72private:

74 // jump information accordingly.
75 void ParseOutfit(const Outfit &outfit);
76 // Add the given distance, cost pair to the jump drive costs and update the fuel cost
77 // of each jump distance if necessary.
78 void UpdateJumpDriveCosts(double distance, double cost);

80

95 bool hasScramDrive = false;

97 bool hasJumpMassCost = false;

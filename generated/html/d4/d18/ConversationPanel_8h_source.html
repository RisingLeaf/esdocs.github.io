41// User interface panel that displays a conversation, allowing you to make
42// choices. If a callback function is given, that function will be called when
43// the panel closes, to report the outcome of the conversation.
44class ConversationPanel : public Panel {
45public:
46 ConversationPanel(PlayerInfo &player, const Conversation &conversation,
47 const Mission *caller = nullptr, const System *system = nullptr,
48 const std::shared_ptr<Ship> &ship = nullptr, bool useTransactions = false);
49
50template <class T>
51 void SetCallback(T *t, void (T::*fun)(int));
52 void SetCallback(std::function<void(int)> fun);
53
54 // Draw this panel.
55 virtual void Draw() override;
56
57
58protected:
59 // Event handlers.
60 virtual bool KeyDown(SDL_Keycode key, Uint16 mod, const Command &command, bool isNewPress) override;
61 virtual bool Drag(double dx, double dy) override;
62 virtual bool Scroll(double dx, double dy) override;
63 virtual bool Hover(int x, int y) override;
64
65
66private:
67 // Go to the given conversation node. If a choice index is given, include
68 // the text of that choice in the conversation history.
69 void Goto(int index, int selectedChoice = -1);
70 // Exit this panel and do whatever needs to happen next, which includes
71 // possibly activating a callback function and, if docked with an NPC,
72 // destroying it or showing the BoardingPanel (if it is hostile).
73 void Exit();
74 // Handle mouse click on the "ok," "done," or a conversation choice.
75 void ClickName(int side);
76 void ClickChoice(int index);
77 // Given an index into the list of displayed choices (i.e. not including
78 // conditionally-skipped choices), return its "raw index" in the
79 // conversation (i.e. including conditionally-skipped choices)
80 int MapChoice(int n) const;
81
82
83private:
84 // Text to be displayed is broken up into chunks (paragraphs). Paragraphs
85 // may also include "scene" images.
86 class Paragraph {
87 public:
88 explicit Paragraph(const std::string &text, const Sprite *scene = nullptr, bool isFirst = false);
89
90 // Get the height of this paragraph.
91 int Height() const;
92 // Get the "center point" of this paragraph. This is for drawing a
93 // highlight under paragraphs that represent choices.
94 Point Center() const;
95 // Draw this paragraph at the given point, and return the point that the
96 // next paragraph below this one should be drawn at.
97 Point Draw(Point point, const Color &color) const;
98
99 private:
100 const Sprite *scene = nullptr;
101 WrappedText wrap;
102 // Special case: if this is the very first paragraph and it begins with
103 // a "scene" image, there is no need for padding above the image.
104 bool isFirst = false;
105 };

108private:
109 // Reference to the player, to apply any changes to them.
110 PlayerInfo &player;
111
112 // A pointer to the mission that called this conversation.
113 const Mission *caller = nullptr;
114
115 // Should we use a PlayerInfo transaction to prevent save-load glitches?
116 bool useTransactions = false;
117
118 // The conversation we are displaying.
119 const Conversation &conversation;
120 // All conversations start with node 0.
121 int node = 0;
122 // This function should be called with the conversation outcome.
123 std::function<void(int)> callback = nullptr;
124
125 // Current scroll position.
126 double scroll = 0.;
127
128 // The "history" of the conversation up to this point:
129 std::list<Paragraph> text;
130 // The current choices being presented to you, and their indices:
131 std::list<std::pair<Paragraph, int>> choices;
132 int choice = 0;
133 // Flicker time, set if the player enters invalid input for a pilot's name.
134 int flickerTime = 0;
135
136 // Text entry fields for changing the player's name.
137 std::string firstName;
138 std::string lastName;
139 // Text substitutions (player's name, and ship name).
140 std::map<std::string, std::string> subs;
141
142 // Maximum scroll amount.
143 double maxScroll = 0.;
144
145 // If specified, this is a star system to display with a special big pointer
146 // when the player brings up the map. (Typically a mission destination.)
147 const System *system;
148 // If specified, this is a pointer to the ship that this conversation
149 // acts upon (e.g. the ship failing a "flight check", or the NPC you
150 // have boarded).
151 std::shared_ptr<Ship> ship;
152
153 // Whether the mouse moved in the current frame.
154 bool isHovering = false;
155 Point hoverPoint;
156};

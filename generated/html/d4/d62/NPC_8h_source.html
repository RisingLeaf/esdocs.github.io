47// An NPC is a ship associated with a mission. Certain required actions are
48// associated with each NPC, such as boarding it, killing it, or making sure it
49// is not boarded or killed. NPCs also have different "behaviors," such as
50// staying in the system they started in, or attacking only the player's ships.
51class NPC {
52public:
53 enum class Trigger : int_fast8_t {
54 // Triggers corresponding directly to ShipEvents.
55 ASSIST,
56 SCAN_CARGO,
57 SCAN_OUTFITS,
58 PROVOKE,
59 DISABLE,
60 BOARD,
61 CAPTURE,
62 DESTROY,
63 ENCOUNTER,
64 // Can be triggered by either the CAPTURE or DESTROY events.
65 KILL,
66 };

69public:
70 NPC() = default;
71 // Copying an NPC instance isn't allowed.

73 NPC &operator=(const NPC &) = delete;
74 NPC(NPC &&) noexcept = default;
75 NPC &operator=(NPC &&) noexcept = default;
76 ~NPC() noexcept = default;

78 // Construct and Load() at the same time.
79 explicit NPC(const DataNode &node);

81 void Load(const DataNode &node);
82 // Note: the Save() function can assume this is an instantiated mission, not
83 // a template, so fleets will be replaced by individual ships already.
84 void Save(DataWriter &out) const;

86 // Determine if this NPC or NPC template uses well-defined data.
87 // Returns the reason the NPC is not valid, or an empty string if valid.
88 std::string Validate(bool asTemplate = false) const;

90 const EsUuid &UUID() const noexcept;

92 // Update or check spawning and despawning for this NPC.
93 void UpdateSpawning(const PlayerInfo &player);
94 bool ShouldSpawn() const;

96 // Get the ships associated with this set of NPCs.
97 const std::list<std::shared_ptr<Ship>> Ships() const;
98
99 // Handle the given ShipEvent.
100 void Do(const ShipEvent &event, PlayerInfo &player, UI *ui = nullptr,
101 const Mission *caller = nullptr, bool isVisible = true);
102 // Determine if the NPC is in a successful state, assuming the player is in the given system.
103 // (By default, a despawnable NPC has succeeded and is not actually checked.)
104 bool HasSucceeded(const System *playerSystem, bool ignoreIfDespawnable = true) const;
105 // Check if the NPC is supposed to be accompanied and is not.
106 bool IsLeftBehind(const System *playerSystem) const;
107 // Determine if the NPC is in a failed state. A failed state is irrecoverable, except for
108 // NPCs which would despawn upon the player's next landing.
109 bool HasFailed() const;

111 // Create a copy of this NPC but with the fleets replaced by the actual
112 // ships they represent, wildcards in the conversation text replaced, etc.

114 int jumps, int64_t payload) const;

140 // Dialog or conversation to show when all requirements for this NPC are met:
141 std::string dialogText;
142 ExclusiveItem<Phrase> dialogPhrase;
143 ExclusiveItem<Conversation> conversation;
144
145 // Conditions that must be met in order for this NPC to be placed or despawned:
146 ConditionSet toSpawn;
147 ConditionSet toDespawn;
148 // Once true, the NPC will be spawned on takeoff and its success state will influence
149 // the parent mission's ability to be completed.
150 bool passedSpawnConditions = false;
151 // Once true, the NPC will be despawned on landing and it will no longer contribute to
152 // the parent mission's ability to be completed or failed.
153 bool passedDespawnConditions = false;

156 bool checkedSpawnConditions = false;

159 // be customized or just refer to stock objects:
160 std::list<std::shared_ptr<Ship>> ships;
161 std::list<const Ship *> stockShips;
162 std::list<std::string> shipNames;
163 std::list<ExclusiveItem<Fleet>> fleets;
164
165 // This must be done to each ship in this set to complete the mission:
166 int succeedIf = 0;
167 int failIf = 0;
168 bool mustEvade = false;
169 bool mustAccompany = false;
170 // The ShipEvent actions that have been done to each ship.
171 std::map<const Ship *, int> shipEvents;
172
173 // The NPCActions that this NPC can run on certain events/triggers.
174 std::map<Trigger, NPCAction> npcActions;

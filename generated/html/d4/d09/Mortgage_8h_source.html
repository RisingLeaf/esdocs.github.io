27// Class representing a mortgage (or a fine that can be paid in installments).
28// The interest rate depends on your credit score, and on whether this is an
29// ordinary mortgage versus a fine (high interest) or a crew bonus (low). This
30// also handles the calculations for determining how big a new mortgage you can
31// qualify for, based on your average daily income.
32class Mortgage {
33public:
34 // Find out how much you can afford to borrow with the given annual revenue
35 // and the given credit score (which should be between 200 and 800).
36 static int64_t Maximum(int64_t annualRevenue, int creditScore, double currentPayments);
37
38
39public:
40 Mortgage() = default;
41 // Create a new mortgage of the given amount. If this is a fine, set the
42 // credit score to zero for a higher interest rate.
43 Mortgage(int64_t principal, int creditScore, int term = 365);
44 // Construct and Load() at the same time.
45 Mortgage(const DataNode &node);
46
47 // Load or save mortgage data.
48 void Load(const DataNode &node);
49 void Save(DataWriter &out) const;
50
51 // Make a mortgage payment. The return value is the amount paid.
52 int64_t MakePayment();
53 void MissPayment();
54 // Pay down additional principal. Unlike a "real" mortgage, this reduces
55 // the minimum amount of your future payments, not the term of the mortgage.
56 // This returns the actual amount paid, which may be less if the total
57 // principal remaining is less than the given amount.
58 int64_t PayExtra(int64_t amount);
59
60 // The type is "Mortgage" if this is a mortgage you applied for from a bank,
61 // and "Fine" if this is a fine imposed on you for illegal activities.
62 const std::string &Type() const;
63 // Get the remaining mortgage principal.
64 int64_t Principal() const;
65 // Get the interest rate. It is formatted as a string, because all that the
66 // program will ever do with this is display it.
67 const std::string &Interest() const;
68 // Get the remaining number of payments that must be made.
69 int Term() const;
70 // Check the amount of the next payment due (rounded to the nearest credit).
71 int64_t Payment() const;
72 // Check the amount of the next payment due.
73 double PrecisePayment() const;
74
75
76private:
77 // Note: once a mortgage is set up, only the principal and term will change.
78 std::string type;
79 int64_t principal = 0;
80 double interest = 0.;
81 std::string interestString;
82 int term = 365;
83};

30// Class for storing a list of textures to blit to the screen. This allows the
31// work of calculating the transformation matrices to be done in a separate
32// thread from the graphics thread. However, the SpriteShader class is also
33// available for drawing individual sprites in contexts where putting them into
34// a DrawList first does not make sense.
35class DrawList {
36public:
37 // Clear the list, also setting the global time step for animation.
38 void Clear(int step = 0, double zoom = 1.);
39 void SetCenter(const Point &center, const Point &centerVelocity = Point());
40
41 // Add an object based on the Body class.
42 bool Add(const Body &body, double cloak = 0.);
43 // Add an object at the given position (rather than its own).
44 bool Add(const Body &body, Point position, double cloak = 0.);
45
46 // Add an object that should not be drawn with motion blur.
47 bool AddUnblurred(const Body &body);
48 // Add an object using a specific swizzle (rather than its own).
49 bool AddSwizzled(const Body &body, int swizzle);
50
51 // Draw all the items in this list.
52 void Draw() const;
53
54
55private:
56 // Determine if the given object should be drawn at all.
57 bool Cull(const Body &body, const Point &position, const Point &blur) const;
58
59 void Push(const Body &body, Point pos, Point blur, double cloak, int swizzle);
60
61
62private:
63 int step = 0;
64 double zoom = 1.;
65 bool isHighDPI = false;
66 std::vector<SpriteShader::Item> items;
67
68 Point center;
69 Point centerVelocity;
70};

41template <class Item>
42void Sale<Item>::Load(const DataNode &node, const Set<Item> &items)
43{
44 for(const DataNode &child : node)
45 {
46 const std::string &token = child.Token(0);
47 bool remove = (token == "clear" || token == "remove");
48 if(remove && child.Size() == 1)
49 this->clear();
50 else if(remove && child.Size() >= 2)
51 this->erase(items.Get(child.Token(1)));
52 else if(token == "add" && child.Size() >= 2)
53 this->insert(items.Get(child.Token(1)));
54 else
55 this->insert(items.Get(token));
56 }
57}

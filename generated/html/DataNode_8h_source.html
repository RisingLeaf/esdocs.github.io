26// A DataNode is a single line of a DataFile. It consists of one or more tokens,
27// which can be interpreted either as strings or as floating point values, and
28// it may also have "children," which may each in turn have their own children.
29// The tokens of a node are separated by white space, with quotation marks being
30// used to group multiple words into a single token. If the token text contains
31// quotation marks, it should be enclosed in backticks instead.
32class DataNode {
33public:
34 // Construct a DataNode. For the purpose of printing stack traces, each node
35 // must remember what its parent node is.
36 explicit DataNode(const DataNode *parent = nullptr) noexcept(false);
37 // Copying or moving a DataNode requires updating the parent pointers.
38 DataNode(const DataNode &other);
39 DataNode &operator=(const DataNode &other);
40 DataNode(DataNode &&) noexcept;
41 DataNode &operator=(DataNode &&) noexcept;
42
43 // Get the number of tokens in this node.
44 int Size() const noexcept;
45 // Get all the tokens in this node as an iterable vector.
46 const std::vector<std::string> &Tokens() const noexcept;
47 // Get the token at the given index. No bounds checking is done internally.
48 // DataFile loading guarantees index 0 always exists.
49 const std::string &Token(int index) const;
50 // Convert the token at the given index to a number. This returns 0 and prints an
51 // error if the index is out of range or the token cannot be interpreted as a number.
52 double Value(int index) const;
53 static double Value(const std::string &token);
54 // Check if the token at the given index is a number in a format that this
55 // class is able to parse.
56 bool IsNumber(int index) const;
57 static bool IsNumber(const std::string &token);
58 // Convert the token at the given index to a boolean. This returns false
59 // and prints an error if the index is out of range or the token cannot
60 // be interpreted as a number.
61 bool BoolValue(int index) const;
62 // Check if the token at the given index is a boolean, i.e. "true"/"1" or "false"/"0"
63 // as a string.
64 bool IsBool(int index) const;
65 static bool IsBool(const std::string &token);
66
67 // Check if this node has any children. If so, the iterator functions below
68 // can be used to access them.
69 bool HasChildren() const noexcept;
70 std::list<DataNode>::const_iterator begin() const noexcept;
71 std::list<DataNode>::const_iterator end() const noexcept;
72
73 // Print a message followed by a "trace" of this node and its parents.
74 int PrintTrace(const std::string &message = "") const;
75
76
77private:
78 // Adjust the parent pointers when a copy is made of a DataNode.
79 void Reparent() noexcept;
80
81
82private:
83 // These are "child" nodes found on subsequent lines with deeper indentation.
84 std::list<DataNode> children;
85 // These are the tokens found in this particular line of the data file.
86 std::vector<std::string> tokens;
87 // The parent pointer is used only for printing stack traces.
88 const DataNode *parent = nullptr;
89 // The line number in the given file that produced this node.
90 size_t lineNumber = 0;
91
92 // Allow DataFile to modify the internal structure of DataNodes.
93 friend class DataFile;
94};
